description: >
  <p>フレームバッファへの図形の描画を開始します。</p>

  <p><code>myBuffer.begin()</code>と<a
  href="/reference/p5.Framebuffer/end/">myBuffer.end()</a>

  を使用すると、フレームバッファに図形を描画できます。<code>myBuffer.begin()</code>は

  フレームバッファへの描画を開始し、

  <a href="/reference/p5.Framebuffer/end/">myBuffer.end()</a>は

  フレームバッファへの描画を停止します。変更は、フレームバッファが

  画像やテクスチャとして表示されるまで見えません。</p>

  - |-

    <div>
    <code>
    let myBuffer;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Framebufferオブジェクトを作成します。
      myBuffer = createFramebuffer();

      describe('空のグレーのキャンバス。ユーザーがマウスを押し続けている間、キャンバスが暗くなり、回転する多色のトーラスが表示されます。');
    }

    function draw() {
      background(200);

      // p5.Framebufferオブジェクトへの描画を開始します。
      myBuffer.begin();

      background(50);
      rotateY(frameCount * 0.01);
      normalMaterial();
      torus(30);

      // p5.Framebufferオブジェクトへの描画を停止します。
      myBuffer.end();

      // ユーザーがマウスを押している間、
      // p5.Framebufferオブジェクトを表示します。
      if (mouseIsPressed === true) {
        image(myBuffer, -50, -50);
      }
    }
