  <p>フレームバッファのカラーデータを格納するオブジェクトです。</p>

  <p>各フレームバッファは内部的に

  <a href="https://developer.mozilla.org/en-US/docs/Web/API/WebGLTexture/"
  target="_blank">WebGLTexture</a>

  オブジェクトを使用してカラーデータを格納します。<code>myBuffer.color</code>
  プロパティは

  このデータを他の関数に直接渡すことを可能にします。例えば、

  <code>texture(myBuffer.color)</code>や

  <code>myShader.setUniform('colorTexture', myBuffer.color)</code>を呼び出すことは、

  高度な使用例に役立つかもしれません。</p>

  <p>注意：デフォルトでは、フレームバッファのy座標は画像やビデオと比べて反転しています。

  必要に応じてフレームバッファのy座標を反転させるのは簡単です。例えば、

  <code>plane(myBuffer.width, -myBuffer.height)</code>を呼び出すと、
  フレームバッファが反転します。</p>

    function setup() {
      createCanvas(100, 100, WEBGL);

      background(200);

      // p5.Framebufferオブジェクトを作成します。
      let myBuffer = createFramebuffer();

      // p5.Framebufferオブジェクトへの描画を開始します。
      myBuffer.begin();

      triangle(-25, 25, 0, -25, 25, 25);

      // p5.Framebufferオブジェクトへの描画を終了します。
      myBuffer.end();

      // p5.FramebufferオブジェクトのWebGLTextureを使用します。
      texture(myBuffer.color);

      // 平面のスタイルを設定します。
      noStroke();

      // 平面を描画します。
      plane(myBuffer.width, myBuffer.height);

      describe('灰色の背景に白い三角形。');
    }
