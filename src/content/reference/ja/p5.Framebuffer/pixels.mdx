description: >
  <p>フレームバッファ内の各ピクセルの色を含む配列です。</p>

  <p><code>myBuffer.pixels</code>配列にアクセスする前に、
  <a href="/reference/p5.Framebuffer/loadPixels/">myBuffer.loadPixels()</a>
  を呼び出す必要があります。
  変更を加えた後は
  <a href="/reference/p5.Framebuffer/updatePixels/">myBuffer.updatePixels()</a>
  を呼び出す必要があります。</p>

  <p>注意：このプロパティを介してピクセルを更新することは、フレームバッファに
  直接描画するよりも遅くなります。<code>myBuffer.pixels</code>をループする代わりに、
  <a href="/reference/p5/p5.Shader">p5.Shader</a>オブジェクトの使用を検討してください。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);

      background(200);

      // p5.Framebufferオブジェクトを作成します。
      let myBuffer = createFramebuffer();

      // pixels配列をロードします。
      myBuffer.loadPixels();

      // フレームバッファの上半分のピクセル数を取得します。
      let numPixels = myBuffer.pixels.length / 2;

      // フレームバッファの上半分をピンク色に設定します。
      for (let i = 0; i < numPixels; i += 4) {
        myBuffer.pixels[i] = 255;
        myBuffer.pixels[i + 1] = 102;
        myBuffer.pixels[i + 2] = 204;
        myBuffer.pixels[i + 3] = 255;
      }

      // pixels配列を更新します。
      myBuffer.updatePixels();

      // p5.Framebufferオブジェクトをキャンバスに描画します。
      image(myBuffer, -50, -50);

      describe('灰色の長方形の上にピンクの長方形があります。');
    }
    </code>
