  <p>フレームバッファを指定された幅と高さにリサイズします。</p>

  <p>パラメータの<code>width</code>と<code>height</code>は、フレームバッファの

  寸法を設定します。例えば、<code>myBuffer.resize(300, 500)</code>を呼び出すと、

  フレームバッファを300×500ピクセルにリサイズし、その後<code>myBuffer.width</code>を300に、

  <code>myBuffer.height</code>を500に設定します。</p>

    <code>
    let myBuffer;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Framebufferオブジェクトを作成します。
      myBuffer = createFramebuffer();

      describe('白い表面上のマルチカラーの球体。ユーザーがマウスを動かすと画像が大きくなったり小さくなったりし、灰色の背景が現れます。');
    }

    function draw() {
      background(200);

      // p5.Framebufferオブジェクトに描画します。
      myBuffer.begin();
      background(255);
      normalMaterial();
      sphere(20);
      myBuffer.end();

      // p5.Framebufferオブジェクトを表示します。
      image(myBuffer, -50, -50);
    }

    // ユーザーがマウスを動かすと
    // p5.Framebufferオブジェクトをリサイズします。
    function mouseMoved() {
      myBuffer.resize(mouseX, mouseY);
    }
