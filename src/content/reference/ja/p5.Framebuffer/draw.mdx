  <p>描画命令を含む関数を呼び出すことで、フレームバッファに描画します。</p>

  <p>パラメータ<code>callback</code>は、フレームバッファへの描画命令を含む関数です。
  例えば、<code>myBuffer.draw(myFunction)</code>を呼び出すと、
  <code>myFunction()</code>という名前の関数がフレームバッファに描画するために
  呼び出されます。これは以下と同じ効果があります：</p>

  <pre><code class="language-js">myBuffer.begin();

  myFunction();

  myBuffer.end();

  </code></pre>

    <div>
    <code>
    // キャンバスをクリックするとフレームバッファが表示されます。

    let myBuffer;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Framebufferオブジェクトを作成します。
      myBuffer = createFramebuffer();

      describe('空のグレーのキャンバス。ユーザーがマウスを押し続けている間、キャンバスが暗くなり、回転する多色のトーラスが表示されます。');
    }

    function draw() {
      background(200);

      // p5.Framebufferオブジェクトに描画します。
      myBuffer.draw(bagel);

      // ユーザーがマウスを押している間、
      // p5.Framebufferオブジェクトを表示します。
      if (mouseIsPressed === true) {
        image(myBuffer, -50, -50);
      }
    }

    // 回転する多色のトーラスを描画します。
    function bagel() {
      background(50);
      rotateY(frameCount * 0.01);
      normalMaterial();
      torus(30);
    }
