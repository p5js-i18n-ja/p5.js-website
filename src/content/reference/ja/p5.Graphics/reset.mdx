  <p>グラフィックスバッファの変換と照明をリセットします。</p>

  <p>デフォルトでは、メインキャンバスは<a href="/reference/p5/draw/">draw()</a>が実行されるたびに

  特定の変換と照明の値をリセットします。<code>p5.Graphics</code>

  オブジェクトは、<code>myGraphics.reset()</code>を呼び出すことで

  これらの値を手動でリセットする必要があります。</p>

    let pg;

    function setup() {
      createCanvas(100, 100);

      // p5.Graphicsオブジェクトを作成します。
      pg = createGraphics(60, 60);

      describe('白い円が暗い正方形の中をゆっくりと下に移動します。ユーザーがマウスを押すと、円は暗い正方形の上部にリセットされます。');
    }

    function draw() {
      background(200);

      // p5.Graphicsオブジェクトの座標系を変換します。
      // 変換は蓄積されるため、白い円が移動します。
      pg.translate(0, 0.1);

      // p5.Graphicsオブジェクトに描画します。
      pg.background(100);
      pg.circle(30, 0, 10);

      // p5.Graphicsオブジェクトを表示します。
      image(pg, 20, 20);

      // メインキャンバスの座標系を変換します。
      // 変換は蓄積されないため、暗い正方形は
      // 常に同じ場所にあります。
      translate(0, 0.1);

      // ユーザーがマウスを押したときに
      // p5.Graphicsオブジェクトをリセットします。
      if (mouseIsPressed === true) {
        pg.reset();
      }
    }

    let pg;

    function setup() {
      createCanvas(100, 100);

      // WebGLモードを使用してp5.Graphicsオブジェクトを作成します。
      pg = createGraphics(100, 100, WEBGL);

      describe("上から赤い光で照らされた球体。ユーザーがマウスをクリックして押し続けている間、球体の表面が光沢を帯びます。");
    }

    function draw() {
      background(200);

      // 右上から赤いポイントライトを追加します。
      pg.pointLight(255, 0, 0, 50, -100, 50);

      // 球体のスタイルを設定します。
      // 照明の値がリセットされると
      // 光沢が現れるはずです。
      pg.noStroke();
      pg.specularMaterial(255);
      pg.shininess(100);

      // 球体を描画します。
      pg.sphere(30);

      // p5.Graphicsオブジェクトを表示します。
      image(pg, -50, -50);

      // ユーザーがマウスを押したときに
      // p5.Graphicsオブジェクトをリセットします。
      if (mouseIsPressed === true) {
        pg.reset();
      }
    }
