    <div class="norender">
    <code>
    // プロジェクトの "assets" フォルダにある
    // "mammals.csv" ファイルが与えられた場合：
    //
    // id,species,name
    // 0,Capra hircus,Goat
    // 1,Panthera pardus,Leopard
    // 2,Equus zebra,Zebra

    let table;

    function preload() {
      //私のテーブルはカンマ区切り値 "csv" で
      //列ラベルを指定するヘッダーがあります
      table = loadTable('/assets/mammals.csv', 'csv', 'header');
    }

    function setup() {
      //指定された正規表現を使用して特定の列を検索し、TableRowオブジェクトを返します
      let mammal = table.matchRow(new RegExp('ant'), 1);
      print(mammal.getString(1));
      //出力 "Panthera pardus"
    }
