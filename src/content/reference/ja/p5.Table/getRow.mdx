    // id,species,name
    // 0,Capra hircus,Goat
    // 1,Panthera pardus,Leopard
    // 2,Equus zebra,Zebra

    let table;

    function preload() {
      //テーブルはカンマ区切り値 "csv" で
      //列ラベルを指定するヘッダーがあります
      table = loadTable('/assets/mammals.csv', 'csv', 'header');
    }

    function setup() {
      let row = table.getRow(1);
      //列ごとに出力します
      //注意：行はオブジェクトであり、配列ではありません
      for (let c = 0; c < table.getColumnCount(); c++) {
        print(row.getString(c));
      }

      describe('画像は表示されません');
    }
