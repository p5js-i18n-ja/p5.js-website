  <p><a href="/reference/p5/addRow/">addRow()</a>を使用して、<a href="/reference/p5/p5.Table">p5.Table</a>オブジェクトに新しいデータ行を追加します。
  デフォルトでは、

  空の行が作成されます。通常、新しい行への参照を

  TableRowオブジェクトに保存し（上の例のnewRowを参照）、

  その後<a href="/reference/p5/set/">set()</a>を使用して個々の値を設定します。</p>

  <p><a href="/reference/p5/p5.TableRow">p5.TableRow</a>オブジェクトがパラメータとして含まれている場合、

  その行が複製されてテーブルに追加されます。</p>

     <code>
     // プロジェクトの "assets" フォルダにある
     // CSV ファイル "mammals.csv" が与えられた場合：
     //
     // id,species,name
     // 0,Capra hircus,Goat
     // 1,Panthera pardus,Leopard
     // 2,Equus zebra,Zebra

     let table;

     function preload() {
     //テーブルはカンマ区切り値 "csv" で
     //列ラベルを指定するヘッダーがあります
     table = loadTable('/assets/mammals.csv', 'csv', 'header');
     }

     function setup() {
     //行を追加
     let newRow = table.addRow();
     newRow.setString('id', table.getRowCount() - 1);
     newRow.setString('species', 'Canis Lupus');
     newRow.setString('name', 'Wolf');

     //結果を表示
     for (let r = 0; r < table.getRowCount(); r++)
       for (let c = 0; c < table.getColumnCount(); c++)
         print(table.getString(r, c));

     describe('画像は表示されません');
     }
