description: >
  <p><code>saveObj()</code>関数は<code>p5.Geometry</code>オブジェクトを
  Wavefront .objファイル形式で3Dモデルとしてエクスポートします。
  これにより、p5.jsで作成した3Dシェイプを他のソフトウェアで
  レンダリング、アニメーション、3Dプリントなどに使用できます。</p>

  <p>エクスポートされた.objファイルには、<code>p5.Geometry</code>の面と頂点、
  およびテクスチャ座標と法線（存在する場合）が含まれます。</p>

    <div>
    <code>
    let myModel;
    let saveBtn;
    function setup() {
      createCanvas(200, 200, WEBGL);
      myModel = buildGeometry(() => {
        for (let i = 0; i < 5; i++) {
          push();
          translate(
            random(-75, 75),
            random(-75, 75),
            random(-75, 75)
          );
          sphere(random(5, 50));
          pop();
        }
      });

      saveBtn = createButton('Save .obj');
      saveBtn.mousePressed(() => myModel.saveObj());

      describe('空間で回転する数個の球体');
    }

    function draw() {
      background(0);
      noStroke();
      lights();
      rotateX(millis() * 0.001);
      rotateY(millis() * 0.002);
      model(myModel);
    }
