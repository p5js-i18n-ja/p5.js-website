  <p>ジオメトリを含む最小のボックスの位置とサイズを計算します。</p>

  <p>バウンディングボックスは、ジオメトリ全体を含む最小の直方体です。各軸に沿った

  ボックスの最小座標と最大座標、およびサイズ（長さ）とオフセット（中心）によって

  定義されます。</p>

  <p><code>myGeometry.calculateBoundingBox()</code>を呼び出すと、バウンディングボックスを

  記述する4つのプロパティを持つオブジェクトが返されます：</p>

  <pre><code class="language-js">// myGeometryのバウンディングボックスを取得します。

  let bbox = myGeometry.calculateBoundingBox();


  // バウンディングボックスをコンソールに出力します。

  console.log(bbox);


  // {

  //  // 各軸に沿った最小座標。

  //  min: { x: -1, y: -2, z: -3 },

  //

  //  // 各軸に沿った最大座標。

  //  max: { x: 1, y: 2, z: 3},

  //

  //  // 各軸に沿ったサイズ（長さ）。

  //  size: { x: 2, y: 4, z: 6},

  //

  //  // 各軸に沿ったオフセット（中心）。

  //  offset: { x: 0, y: 0, z: 0}

  // }

  </code></pre>

    let particles;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // ランダムな球体を持つ新しいp5.Geometryオブジェクトを作成します。
      particles = buildGeometry(createParticles);

      describe('灰色の背景に対してランダムに配置された10個の白い球体。球体を囲むボックスがあります。');
    }

    function draw() {
      background(50);

      // マウスでの軌道制御を有効にします。
      orbitControl();

      // ライトをオンにします。
      lights();

      // パーティクルのスタイルを設定します。
      noStroke();
      fill(255);

      // パーティクルを描画します。
      model(particles);

      // バウンディングボックスを計算します。
      let bbox = particles.calculateBoundingBox();

      // バウンディングボックスの中心に移動します。
      translate(bbox.offset.x, bbox.offset.y, bbox.offset.z);

      // バウンディングボックスのスタイルを設定します。
      stroke(255);
      noFill();

      // バウンディングボックスを描画します。
      box(bbox.size.x, bbox.size.y, bbox.size.z);
    }

    function createParticles() {
      for (let i = 0; i < 10; i += 1) {
        // ランダムな座標を計算します。
        let x = randomGaussian(0, 15);
        let y = randomGaussian(0, 15);
        let z = randomGaussian(0, 15);

        push();
        // パーティクルの座標に移動します。
        translate(x, y, z);
        // パーティクルを描画します。
        sphere(3);
        pop();
      }
    }
