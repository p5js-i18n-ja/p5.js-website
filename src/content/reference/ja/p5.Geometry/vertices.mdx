description: >
  <p>ジオメトリの頂点を含む配列。</p>

  <p>ジオメトリの頂点は

  <a href="/reference/p5/p5.Vector">p5.Vector</a>オブジェクトとして
  <code>myGeometry.vertices</code>

  配列に格納されています。ジオメトリの最初の頂点は

  <a href="/reference/p5/p5.Vector">p5.Vector</a>オブジェクトで
  <code>myGeometry.vertices[0]</code>、

  2番目の頂点は<code>myGeometry.vertices[1]</code>、3番目の頂点は

  <code>myGeometry.vertices[2]</code>、というように続きます。</p>

    <code>
    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。

    let myGeometry;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Geometryオブジェクトを作成します。
      myGeometry = new p5.Geometry();

      // 頂点の位置を決めるp5.Vectorオブジェクトを作成します。
      let v0 = createVector(-40, 0, 0);
      let v1 = createVector(0, -40, 0);
      let v2 = createVector(40, 0, 0);

      // p5.Geometryオブジェクトのvertices配列に頂点を追加します。
      myGeometry.vertices.push(v0, v1, v2);

      describe('灰色の背景に描かれた白い三角形。');
    }

    function draw() {
      background(200);

      // マウスでの軌道制御を有効にします。
      orbitControl();

      // p5.Geometryオブジェクトを描画します。
      model(myGeometry);
    }

    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。

    let myGeometry;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Geometryオブジェクトを作成します。
      beginGeometry();
      torus(30, 15, 10, 8);
      myGeometry = endGeometry();

      describe('暗い灰色の背景に対してゆっくりと回転する白いトーラス。赤い球体がその頂点を示しています。');
    }

    function draw() {
      background(50);

      // マウスでの軌道制御を有効にします。
      orbitControl();

      // ライトをオンにします。
      lights();

      // 座標系を回転させます。
      rotateY(frameCount * 0.01);

      // p5.Geometryオブジェクトのスタイルを設定します。
      fill(255);
      stroke(0);

      // p5.Geometryオブジェクトを表示します。
      model(myGeometry);

      // 頂点のスタイルを設定します。
      fill(255, 0, 0);
      noStroke();

      // vertices配列をイテレートします。
      for (let v of myGeometry.vertices) {
        // 頂点を示す球体を描画します。
        push();
        translate(v);
        sphere(2.5);
        pop();
      }
    }
