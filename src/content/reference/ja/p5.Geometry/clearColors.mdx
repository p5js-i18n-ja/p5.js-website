  <p>ジオメトリの内部カラーを削除します。</p>

  <p><code>p5.Geometry</code>オブジェクトは、頂点や形状全体に割り当てられた
  「内部カラー」で作成できます。ジオメトリに内部カラーがある場合、
  <a href="/reference/p5/fill/">fill()</a>は効果がありません。
  <code>myGeometry.clearColors()</code>を呼び出すことで、
  <a href="/reference/p5/fill/">fill()</a>関数がジオメトリに色を適用できるようになります。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);

      background(200);

      // p5.Geometryオブジェクトを作成します。
      // 内部カラーを赤に設定します。
      beginGeometry();
      fill(255, 0, 0);
      plane(20);
      let myGeometry = endGeometry();

      // 形状のスタイルを設定します。
      noStroke();

      // p5.Geometryオブジェクトを描画します（中央）。
      model(myGeometry);

      // 原点を右下に移動します。
      translate(25, 25, 0);

      // ジオメトリを緑で塗りつぶそうとします。
      fill(0, 255, 0);

      // ジオメトリを再度描画します（右下）。
      model(myGeometry);

      // ジオメトリの色をクリアします。
      myGeometry.clearColors();

      // ジオメトリを青で塗りつぶします。
      fill(0, 0, 255);

      // 原点を上に移動します。
      translate(0, -50, 0);

      // ジオメトリを再度描画します（右上）。
      model(myGeometry);

      describe(
        '灰色の背景に3つの四角形が描かれています。赤い四角形が中央と右下にあります。青い四角形が右上にあります。'
      );
    }
    </code>
