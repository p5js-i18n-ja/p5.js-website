description: >
  <p>ジオメトリの頂点のうち、どれがそれぞれの面を形成するかをリストした配列です。</p>

  <p>すべての3D形状は、<em>頂点</em>と呼ばれる点の集合を接続して作られます。
  ジオメトリの表面は、頂点を接続して三角形を形成し、それらを縫い合わせることで
  作られます。ジオメトリの表面上の各三角形のパッチを<em>面</em>と呼びます。</p>

  <p>ジオメトリの頂点は、
  <a href="/reference/p5.Geometry/vertices/">myGeometry.vertices</a>配列内に
  <a href="/reference/p5/p5.Vector">p5.Vector</a>オブジェクトとして格納されています。
  ジオメトリの最初の頂点は<code>myGeometry.vertices[0]</code>にある
  <a href="/reference/p5/p5.Vector">p5.Vector</a>オブジェクトで、2番目の頂点は
  <code>myGeometry.vertices[1]</code>、3番目の頂点は<code>myGeometry.vertices[2]</code>
  というように続きます。</p>

  <p>例えば、長方形から作られたジオメトリは2つの面を持ちます。なぜなら、長方形は
  2つの三角形を結合して作られるからです。長方形の<code>myGeometry.faces</code>は
  <code>[[0, 1, 2], [2, 1, 3]]</code>という2次元配列になります。
  最初の面、<code>myGeometry.faces[0]</code>は<code>[0, 1, 2]</code>という配列です。
  これは<code>myGeometry.vertices[0]</code>、<code>myGeometry.vertices[1]</code>、
  <code>myGeometry.vertices[2]</code>を接続して形成されるからです。
  2番目の面、<code>myGeometry.faces[1]</code>は<code>[2, 1, 3]</code>という配列です。
  これは<code>myGeometry.vertices[2]</code>、<code>myGeometry.vertices[1]</code>、
  <code>myGeometry.vertices[3]</code>を接続して形成されるからです。</p>

    <div>
    <code>
    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。

    let myGeometry;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Geometryオブジェクトを作成します。
      beginGeometry();
      sphere();
      myGeometry = endGeometry();

      describe("灰色の背景に描かれた球体。球体の表面は三角形のグレースケールのパッチワークになっています。");
    }

    function draw() {
      background(200);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // ライトをオンにします。
      lights();

      // p5.Geometryオブジェクトのスタイルを設定します。
      noStroke();

      // ランダムシードを設定します。
      randomSeed(1234);

      // faces配列をイテレートします。
      for (let face of myGeometry.faces) {

        // 面のスタイルを設定します。
        let g = random(0, 255);
        fill(g);

        // 面を描画します。
        beginShape();
        // 面を形成する頂点をイテレートします。
        for (let f of face) {
          // 頂点のp5.Vectorオブジェクトを取得します。
          let v = myGeometry.vertices[f];
          vertex(v.x, v.y, v.z);
        }
        endShape();

      }
    }
