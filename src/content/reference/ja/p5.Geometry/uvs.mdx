description: >
  <p>ジオメトリの各頂点のテクスチャ座標をリストした配列です。</p>

  <p><a href="/reference/p5/texture/">texture()</a>を機能させるために、ジオメトリは
  その表面上の点をテクスチャとして使用される長方形の画像のピクセルに
  マッピングする方法が必要です。座標<code>(x, y, z)</code>のジオメトリの頂点は、
  座標<code>(u, v)</code>のテクスチャ画像のピクセルにマッピングされます。</p>

  <p><code>myGeometry.uvs</code>配列は、ジオメトリに追加された順序で各頂点の
  <code>(u, v)</code>座標を格納します。例えば、最初の頂点<code>myGeometry.vertices[0]</code>の
  <code>(u, v)</code>座標は<code>myGeometry.uvs[0]</code>と<code>myGeometry.uvs[1]</code>に
  格納されます。</p>

    let img;

    // 画像を読み込み、p5.Imageオブジェクトを作成します。
    function preload() {
      img = loadImage('/assets/laDefense.jpg');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      background(200);

      // p5.Geometryオブジェクトを作成します。
      let geom1 = buildGeometry(createShape);
      let geom2 = buildGeometry(createShape);

      // 左（オリジナル）
      push();
      translate(-25, 0, 0);
      texture(img);
      noStroke();
      model(geom1);
      pop();

      // geom2のテクスチャ座標を設定します。
      geom2.uvs = [0.25, 0.25, 0.75, 0.25, 0.25, 0.75, 0.75, 0.75];

      // 右（ズームイン）
      push();
      translate(25, 0, 0);
      texture(img);
      noStroke();
      model(geom2);
      pop();

      describe(
        '灰色の背景に2つの天井の写真。右側の写真は写真の中央部分にズームインしています。'
      );
    }

    function createShape() {
      plane(40);
    }
