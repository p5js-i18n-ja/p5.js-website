description: >
  <p>ジオメトリの頂点を変換して、原点を中心とする100×100×100のボックス内にぴったりと収まるようにします。</p>

  <p><code>myGeometry.normalize()</code>を呼び出すと、ジオメトリの頂点が変換され、
  原点<code>(0, 0, 0)</code>を中心に配置されます。その後、頂点がスケーリングされ、
  100×100×100のボックスを埋めるようになります。結果として、小さなジオメトリは
  大きくなり、大きなジオメトリは小さくなります。</p>

  <p>注意：<code>myGeometry.normalize()</code>は
  <a href="/reference/p5/setup/">setup()</a>関数内で呼び出された場合にのみ機能します。</p>

    <div>
    <code>
    let myGeometry;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 非常に小さなトーラスを作成します。
      beginGeometry();
      torus(1, 0.25);
      myGeometry = endGeometry();

      // トーラスを正規化して、その頂点が
      // [-100, 100]の範囲を埋めるようにします。
      myGeometry.normalize();

      describe('白いトーラスが暗い灰色の背景に対してゆっくりと回転します。');
    }

    function draw() {
      background(50);

      // ライトをオンにします。
      lights();

      // y軸を中心に回転します。
      rotateY(frameCount * 0.01);

      // トーラスのスタイルを設定します。
      noStroke();

      // トーラスを描画します。
      model(myGeometry);
    }
