    let cam;

    // フォントを読み込み、p5.Fontオブジェクトを作成します。
    function preload() {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Cameraオブジェクトを作成します。
      cam = createCamera();

      // カメラを右上に配置します。
      cam.setPosition(100, -400, 800);

      // カメラを(10, 20, -30)に向けます。
      cam.lookAt(10, 20, -30);

      describe(
        'グレーの背景に白い立方体。カメラの焦点が移動するにつれて、立方体が上下に動くように見えます。立方体の下に黒字で "centerY: Y" と書かれています。Yは-5から45の間で振動します。'
      );
    }

    function draw() {
      background(200);

      // ボックスのスタイルを設定します。
      fill(255);

      // ボックスを描画します。
      box();

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);
      textFont(font);
      fill(0);

      // 新しいy座標を計算します。
      let y = 25 * sin(frameCount * 0.01) + 20;

      // カメラを向けます。
      cam.lookAt(10, y, -30);

      // centerYの値をもっとも近い整数に丸めて表示します。
      text(`centerY: ${round(cam.centerY)}`, 0, 55);
    }
