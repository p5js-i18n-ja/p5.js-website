    <div>
    <code>
    // ダブルクリックでカメラのズームを「リセット」します。

    let cam1;
    let cam2;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 最初のカメラを作成します。
      cam1 = createCamera();

      // カメラを右上に配置します。
      cam1.setPosition(400, -400, 800);

      // 原点を向くようにします。
      cam1.lookAt(0, 0, 0);

      // 2番目のカメラを作成します。
      cam2 = createCamera();

      // cam1の設定をコピーします。
      cam2.set(cam1);

      describe(
        '灰色の背景に白い立方体が描かれています。カメラがゆっくりと前進します。ユーザーがダブルクリックするとカメラがリセットされます。'
      );
    }

    function draw() {
      background(200);

      // cam2の位置を更新します。
      cam2.move(0, 0, -1);

      // 箱を描画します。
      box();
    }

    // ユーザーがダブルクリックしたときにカメラを「リセット」します。
    function doubleClicked() {
      cam2.set(cam1);
    }
