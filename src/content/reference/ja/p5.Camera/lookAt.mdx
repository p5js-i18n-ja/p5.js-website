description: >
  <p>カメラを特定の位置に向けます。</p>

  <p><code>myCamera.lookAt()</code>は、カメラの位置を変えずに、
  カメラの向きを変更します。</p>

  <p>パラメータ<code>x</code>、<code>y</code>、<code>z</code>は、カメラが向けられるべき
  「ワールド」空間の座標です。例えば、
  <code>myCamera.lookAt(10, 20, 30)</code>を呼び出すと、カメラは座標
  <code>(10, 20, 30)</code>に向けられます。</p>

    <div>
    <code>
    // ダブルクリックで異なるキューブを見ます。

    let cam;
    let isLookingLeft = true;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Cameraオブジェクトを作成します。
      cam = createCamera();

      // カメラを上部中央に配置します。
      cam.setPosition(0, -400, 800);

      // カメラを原点に向けます。
      cam.lookAt(-30, 0, 0);

      describe(
        '灰色の背景に赤いキューブと青いキューブがあります。ユーザーがダブルクリックするとカメラの焦点がキューブ間で切り替わります。'
      );
    }

    function draw() {
      background(200);

      // 左側のボックスを描画します。
      push();
      // 原点を左に移動します。
      translate(-30, 0, 0);
      // ボックスのスタイルを設定します。
      fill(255, 0, 0);
      // ボックスを描画します。
      box(20);
      pop();

      // 右側のボックスを描画します。
      push();
      // 原点を右に移動します。
      translate(30, 0, 0);
      // ボックスのスタイルを設定します。
      fill(0, 0, 255);
      // ボックスを描画します。
      box(20);
      pop();
    }

    // ユーザーがダブルクリックしたときにカメラの焦点を変更します。
    function doubleClicked() {
      if (isLookingLeft === true) {
        cam.lookAt(30, 0, 0);
        isLookingLeft = false;
      } else {
        cam.lookAt(-30, 0, 0);
        isLookingLeft = true;
      }
    }
