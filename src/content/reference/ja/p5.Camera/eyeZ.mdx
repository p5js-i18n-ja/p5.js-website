    let cam;
    let font;

    // フォントを読み込み、p5.Fontオブジェクトを作成します。
    function preload() {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Cameraオブジェクトを作成します。
      cam = createCamera();

      // カメラを上部中央に配置します。
      cam.setPosition(0, -400, 800);

      // カメラを原点に向けます。
      cam.lookAt(0, 0, 0);

      describe(
        'グレーの背景に白い立方体。カメラが動くにつれて、立方体が前後に動くように見えます。立方体の下に黒字で "eyeZ: Z" と書かれています。Zは700から900の間で振動します。'
      );
    }

    function draw() {
      background(200);

      // ボックスのスタイルを設定します。
      fill(255);

      // ボックスを描画します。
      box();

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);
      textFont(font);
      fill(0);

      // 新しいz座標を計算します。
      let z = 100 * sin(frameCount * 0.01) + 800;

      // カメラの位置を設定します。
      cam.setPosition(0, -400, z);

      // eyeZの値を最も近い整数に丸めて表示します。
      text(`eyeZ: ${round(cam.eyeZ)}`, 0, 55);
    }
