  <p>カメラの向きを変えずに、その「ローカル」軸に沿って移動させます。</p>

  <p>パラメータ<code>x</code>、<code>y</code>、<code>z</code>は、カメラが移動すべき距離です。

  例えば、<code>myCamera.move(10, 20, 30)</code>を呼び出すと、カメラは「ローカル」

  空間で右に10ピクセル、下に20ピクセル、後ろに30ピクセル移動します。</p>

  - |-

    <div>
    <code>
    // キャンバスをクリックしてキー押下の検出を開始します。

    let cam;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 最初のカメラを作成します。
      // デフォルト設定のままにします。
      cam = createCamera();

      // カメラを右上に配置します。
      cam.setPosition(400, -400, 800);

      // 原点を向くようにします。
      cam.lookAt(0, 0, 0);

      describe(
        'グレーの背景に白い立方体が描かれています。ユーザーが特定のキーを押すと、立方体が動いているように見えます。'
      );
    }

    function draw() {
      background(200);

      // ユーザーが特定のキーを押したときに
      // カメラを「ローカル」軸に沿って移動させます。
      if (keyIsPressed === true) {

        // 水平方向に移動。
        if (keyCode === LEFT_ARROW) {
          cam.move(-1, 0, 0);
        }
        if (keyCode === RIGHT_ARROW) {
          cam.move(1, 0, 0);
        }

        // 垂直方向に移動。
        if (keyCode === UP_ARROW) {
          cam.move(0, -1, 0);
        }
        if (keyCode === DOWN_ARROW) {
          cam.move(0, 1, 0);
        }

        // 画面の奥行き方向に移動。
        if (key === 'i') {
          cam.move(0, 0, -1);
        }
        if (key === 'o') {
          cam.move(0, 0, 1);
        }
      }

      // 箱を描画します。
      box();
    }
