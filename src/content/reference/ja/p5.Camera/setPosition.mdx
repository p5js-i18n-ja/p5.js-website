description: >
  <p>カメラの向きを変えずに、「ワールド」空間内でのカメラの位置を設定します。</p>

  <p>パラメータ<code>x</code>、<code>y</code>、<code>z</code>は、カメラを配置する座標です。

  例えば、<code>myCamera.setPosition(10, 20, 30)</code>を呼び出すと、

  カメラは「ワールド」空間の座標<code>(10, 20, 30)</code>に配置されます。</p>

    <div>
    <code>
    // ダブルクリックでカメラを切り替えます。

    let cam1;
    let cam2;
    let isDefaultCamera = true;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 最初のカメラを作成します。
      // デフォルト設定のままにします。
      cam1 = createCamera();

      // 2番目のカメラを作成します。
      cam2 = createCamera();

      // 原点に近づけて配置します。
      cam2.setPosition(0, 0, 600);

      // 現在のカメラをcam1に設定します。
      setCamera(cam1);

      describe(
        '灰色の背景に白い立方体の列。ユーザーがダブルクリックするとカメラのズーム量が切り替わります。'
      );
    }

    function draw() {
      background(200);

      // 原点をカメラに向かって移動します。
      translate(-10, 10, 500);

      // 座標系を回転させます。
      rotateY(-0.1);
      rotateX(-0.1);

      // 立方体の列を描画します。
      for (let i = 0; i < 6; i += 1) {
        translate(0, 0, -30);
        box(10);
      }
    }

    // ユーザーがダブルクリックしたときに現在のカメラを切り替えます。
    function doubleClicked() {
      if (isDefaultCamera === true) {
        setCamera(cam2);
        isDefaultCamera = false;
      } else {
        setCamera(cam1);
        isDefaultCamera = true;
      }
    }
