description: >
  <p>カメラを左右に回転させます。</p>

  <p>パンはカメラの位置を変えずに回転させます。回転は

  カメラの「ローカル」空間で行われます。</p>

  <p>パラメータ<code>angle</code>は、カメラが回転すべき角度です。
  正の角度を渡すと、例えば<code>myCamera.pan(0.001)</code>のように、カメラは

  右に回転します。負の角度を渡すと、例えば<code>myCamera.pan(-0.001)</code>のように、

  カメラは左に回転します。</p>

  <p>注意：角度は現在の<a href="/reference/p5/angleMode/">angleMode()</a>に基づいて解釈されます。</p>

    <div>
    <code>
    let cam;
    let delta = 0.001;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Cameraオブジェクトを作成します。
      cam = createCamera();

      // カメラを上部中央に配置します。
      cam.setPosition(0, -400, 800);

      // カメラを原点に向けます。
      cam.lookAt(0, 0, 0);

      describe(
        'グレーの背景に白い立方体。カメラが左右にパンするにつれて、立方体が視界に入ったり出たりします。'
      );
    }

    function draw() {
      background(200);

      // カメラでパンします。
      cam.pan(delta);

      // 120フレームごとに方向を切り替えます。
      if (frameCount % 120 === 0) {
        delta *= -1;
      }

      // 箱を描画します。
      box();
    }
