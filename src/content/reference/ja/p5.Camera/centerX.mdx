      createCanvas(100, 100, WEBGL);

      cam = createCamera();

      // カメラを右上に配置します。
      cam.setPosition(100, -400, 800);

      // カメラを(10, 20, -30)に向けます。
      cam.lookAt(10, 20, -30);

      describe(
        'グレーの背景に白い立方体。カメラの焦点が移動すると、立方体が左右に動くように見えます。立方体の下に黒字で "centerX: X" と書かれています。Xは-15から35の間で振動します。'
      );
    }

    function draw() {
      background(200);

      // ボックスのスタイルを設定します。
      fill(255);

      // ボックスを描画します。
      box();

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);
      textFont(font);
      fill(0);

      // 新しいx座標を計算します。
      let x = 25 * sin(frameCount * 0.01) + 10;

      // カメラを向けます。
      cam.lookAt(x, 20, -30);

      // centerXの値を最も近い整数に丸めて表示します。
      text(`centerX: ${round(cam.centerX)}`, 0, 55);
    }
