description: >
  <p>要素上でマウスホイールがスクロールされたときに関数を呼び出します。</p>

  <p>コールバック関数 <code>fxn</code> には <code>event</code> オブジェクトが渡されます。
  <code>event</code> には2つの数値プロパティ、<code>deltaY</code> と <code>deltaX</code> があります。
  <code>event.deltaY</code> は、マウスホイールがユーザーから離れる方向に回転した場合は負の値になります。
  ユーザーに向かって回転した場合は正の値になります。<code>event.deltaX</code> は、
  マウスホイールが右に移動した場合は正の値になります。左に移動した場合は負の値になります。</p>

  <p><code>myElement.mouseWheel(false)</code> を呼び出すと、この関数は無効になります。</p>

    function setup() {
      // キャンバス要素を作成し、
      // cnvに割り当てます。
      let cnv = createCanvas(100, 100);

      background(200);

      // マウスホイールが動いたときに
      // changeBackground()を呼び出します。
      cnv.mouseWheel(changeBackground);

      describe('灰色の正方形。マウスホイールが正方形上でスクロールされると、色が変わり図形が表示されます。');
    }

    function changeBackground(event) {
      // deltaYに基づいて
      // 背景色を変更します。
      if (event.deltaY > 0) {
        background('deeppink');
      } else if (event.deltaY < 0) {
        background('cornflowerblue');
      } else {
        background(200);
      }

      // deltaXに基づいて図形を描画します。
      if (event.deltaX > 0) {
        circle(50, 50, 20);
      } else if (event.deltaX < 0) {
        square(40, 40, 20);
      }
    }
