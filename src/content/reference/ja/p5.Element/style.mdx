description: >
  <p>要素に<a href="https://developer.mozilla.org/en-US/docs/Web/CSS/Syntax/"
  target="_blank">CSS宣言</a>を追加することでスタイルを適用します。</p>

  <p>最初のパラメータ<code>property</code>は文字列です。<code>myElement.style('color')</code>のように
  スタイルプロパティの名前が渡された場合、メソッドは現在の値を文字列として返すか、
  設定されていない場合は<code>null</code>を返します。<code>myElement.style('color:deeppink')</code>のように
  <code>property:style</code>文字列が渡された場合、メソッドはスタイル<code>property</code>を
  <code>value</code>に設定します。</p>

  <p>2番目のパラメータ<code>value</code>はオプションです。プロパティの値を設定します。
  <code>value</code>は<code>myElement.style('color', 'deeppink')</code>のように文字列、
  または<code>myElement.style('color', myColor)</code>のように
  <a href="/reference/p5/p5.Color">p5.Color</a>オブジェクトにできます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 空の段落要素を作成し、フォントの色を"deeppink"に設定します。
      let p = createP();
      p.position(5, 5);
      p.style('color', 'deeppink');

      // 要素の色をRGBカラー文字列として取得します。
      let c = p.style('color');

      // RGBカラー文字列を使用して要素の内部HTMLを設定します。
      p.html(c);

      describe('灰色の背景にピンク色で書かれたテキスト"rgb(255, 20, 147)"。');
    }
