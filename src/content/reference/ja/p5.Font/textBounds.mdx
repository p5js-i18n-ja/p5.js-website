description: >
  <p>指定されたテキスト文字列の境界ボックスを計算します。</p>

  <p>境界ボックスは、テキストを含む最小の長方形です。
  これは、テキストの配置や衝突検出に役立ちます。</p>

  <p>返されるオブジェクトには以下のプロパティがあります：</p>

  <ul>

  <li><code>x</code>：境界ボックスの左上隅のx座標。</li>

  <li><code>y</code>：境界ボックスの左上隅のy座標。</li>

  <li><code>w</code>：境界ボックスの幅。</li>

  <li><code>h</code>：境界ボックスの高さ。</li>

  </ul>
line: 170
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let myFont;
    let bbox;

    function preload() {
      myFont = loadFont('/assets/inconsolata.otf');
    }

    function setup() {
      createCanvas(100, 100);
      background(200);

      textFont(myFont);
      textSize(20);

      let message = 'こんにちは';
      bbox = myFont.textBounds(message, 0, 0, 20);

      // テキストを描画します。
      fill(0);
      text(message, 10, 30);

      // 境界ボックスを描画します。
      noFill();
      stroke(255, 0, 0);
      rect(10 + bbox.x, 30 + bbox.y, bbox.w, bbox.h);

      describe('灰色の背景に「こんにちは」と書かれています。テキストの周りに赤い長方形が描かれています。');
    }
    </code>
    </div>
