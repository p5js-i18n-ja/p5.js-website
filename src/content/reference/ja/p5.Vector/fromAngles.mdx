      background(0);

      // ISO角を計算します。
      let theta = frameCount *  0.05;
      let phi = 0;

      // p5.Vectorオブジェクトを作成します。
      let v = p5.Vector.fromAngles(theta, phi, 100);

      // p5.Vectorを使用してポイントライトを作成します。
      let c = color('deeppink');
      pointLight(c, v);

      // 球体のスタイルを設定します。
      fill(255);
      noStroke();

      // 球体を描画します。
      sphere(35);
    }
