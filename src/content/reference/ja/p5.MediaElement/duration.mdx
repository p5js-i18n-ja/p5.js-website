      let s = dragon.duration() - dragon.time();

      // 表示用にsを小数点以下1桁に丸めます。
      s = round(s, 1);

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);

      // 残り時間を表示します。
      text(`${s} 秒残り`, 50, 50);
