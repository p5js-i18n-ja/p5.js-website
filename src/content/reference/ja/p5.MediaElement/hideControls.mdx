    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER);

      // コントロールが非表示または表示されているときに異なるメッセージを表示します。
      if (isHidden === true) {
        text('ダブルクリックでコントロールを表示', 10, 20, 80, 80);
      } else {
        text('ダブルクリックでコントロールを非表示', 10, 20, 80, 80);
      }
    }

    // ダブルクリックに基づいてコントロールを表示/非表示にします。
    function doubleClicked() {
      if (isHidden === true) {
        dragon.showControls();
        isHidden = false;
      } else {
        dragon.hideControls();
        isHidden = true;
      }
