    let beat;
    let isPlaying = false;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // createAudio()を使用してp5.MediaElementを作成します。
      beat = createAudio('/assets/beat.mp3');

      describe('灰色の背景に黒で「クリックして再生」と書かれたテキスト。ユーザーがクリックするとビートが再生されます。ユーザーが再度クリックするとビートが停止します。');
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);

      // 再生状態に基づいて異なる指示を表示します。
      if (isPlaying === true) {
        text('クリックして停止', 50, 50);
      } else {
        text('クリックして再生', 50, 50);
      }
    }

    // ユーザーがマウスを押したときに再生を調整します。
    function mousePressed() {
      if (isPlaying === true) {
        // ビートが再生中の場合、停止します。
        beat.stop();
        isPlaying = false;
      } else {
        // ビートが停止中の場合、再生します。
        beat.play();
        isPlaying = true;
      }
    }
