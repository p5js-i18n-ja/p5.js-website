    let beat;
    let isLooping = false;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // createAudio()を使用してp5.MediaElementを作成します。
      beat = createAudio('/assets/beat.mp3');

      describe('灰色の背景に黒で書かれた「クリックしてループ」というテキスト。ユーザーがクリックするとビートが繰り返しループ再生されます。再度クリックするとビートが停止します。');
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);

      // 再生状態に基づいて異なる指示を表示します。
      if (isLooping === true) {
        text('クリックして停止', 50, 50);
      } else {
        text('クリックしてループ', 50, 50);
      }
    }

    // ユーザーがマウスを押したときに再生を調整します。
    function mousePressed() {
      if (isLooping === true) {
        // ビートがループ中なら停止します。
        beat.stop();
        isLooping = false;
      } else {
        // ビートが停止中ならループを開始します。
        beat.loop();
        isLooping = true;
      }
    }
