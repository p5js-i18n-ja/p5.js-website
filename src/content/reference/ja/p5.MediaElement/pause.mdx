    let beat;
    let isPaused = true;

    function setup() {
      createCanvas(100, 100);

      // createAudio()を使用してp5.MediaElementを作成します。
      beat = createAudio('/assets/beat.mp3');

      describe('灰色の背景に黒で書かれた「クリックして再生」というテキスト。ユーザーが四角をクリックするとビートが再生または一時停止します。');
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);

      // 再生状態に基づいて異なる指示を表示します。
      if (isPaused === true) {
        text('クリックして再生', 50, 50);
      } else {
        text('クリックして一時停止', 50, 50);
      }
    }

    // ユーザーがマウスを押したときに再生を調整します。
    function mousePressed() {
      if (isPaused === true) {
        // ビートが一時停止している場合、
        // 再生します。
        beat.play();
        isPaused = false;
      } else {
        // ビートが再生中の場合、
        // 一時停止します。
        beat.pause();
        isPaused = true;
      }
    }
