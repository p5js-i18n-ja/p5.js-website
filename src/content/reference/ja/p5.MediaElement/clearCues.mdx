    let isChanging = true;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // createAudio()を使用してp5.MediaElementを作成します。
      let beat = createAudio('/assets/beat.mp3');

      // ビートをループ再生します。
      beat.loop();

      // いくつかのイベントをスケジュールします。
      beat.addCue(0, changeBackground, 'red');
      beat.addCue(2, changeBackground, 'deeppink');
      beat.addCue(4, changeBackground, 'orchid');
      beat.addCue(6, changeBackground, 'lavender');

      describe('正方形の上に「ダブルクリックで変更を停止」と書かれたテキスト。オーディオの再生中、2秒ごとに色が変わります。ユーザーが正方形をダブルクリックすると、色の変更が停止します。');
    }

    function draw() {
      background(200);

      // 利用可能なコールバックに基づいて異なる指示を表示します。
      if (isChanging === true) {
        text('ダブルクリックで変更を停止。', 10, 10, 80, 80);
      } else {
        text('これ以上変更はありません。', 10, 10, 80, 80);
      }
    }

    // 背景色を変更します。
    function changeBackground(c) {
      background(c);
    }

    // ユーザーがダブルクリックしたときに、キューに入れられた関数を削除し、
    // 色の変更を停止します。
    function doubleClicked() {
      if (isChanging === true) {
        beat.clearCues();
        isChanging = false;
      }
