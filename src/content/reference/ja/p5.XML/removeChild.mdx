  <p>最初に一致する子要素を削除します。</p>

  <p>パラメータ<code>name</code>は、削除する子要素です。文字列が渡された場合、
  例えば<code>myXML.removeChild('cat')</code>のように、
  <code><cat></code>タグを持つ最初の子要素が削除されます。数値が渡された場合、
  例えば<code>myXML.removeChild(1)</code>のように、その索引の子要素が
  削除されます。</p>

    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のmammal要素を削除します。
      myXML.removeChild('mammal');

      // 子要素の配列を取得します。
      let children = myXML.getChildren();

      // テキストのスタイルを設定します。
      textAlign(LEFT, CENTER);
      textFont('Courier New');
      textSize(14);

      // 配列を反復処理します。
      for (let i = 0; i < children.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 25;

        // 子要素のコンテンツを取得します。
        let content = children[i].getContent();

        // 子要素のコンテンツを表示します。
        text(content, 10, y);
      }

      describe(
        '「Leopard」、「Zebra」、「Turtle」という単語が別々の行に書かれています。テキストは灰色の背景に黒で表示されています。'
      );
    }
