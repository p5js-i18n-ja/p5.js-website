description: >
  <p>要素のタグ名を設定します。</p>

  <p>XML要素の名前はそのタグによって与えられます。例えば、

  <code><language>JavaScript</language></code>という要素の名前は
  <code>language</code>です。</p>

  <p>パラメータ<code>name</code>は、要素の新しい名前を文字列で指定します。
  例えば、

  <code>myXML.setName('planet')</code>を呼び出すと、要素の新しいタグ名は

  <code><planet></planet></code>になります。</p>

example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 要素の元の名前を取得します。
      let oldName = myXML.getName();

      // 要素の名前を設定します。
      myXML.setName('monsters');

      // 要素の新しい名前を取得します。
      let newName = myXML.getName();

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素の名前を表示します。
      text(oldName, 50, 33);
      text(newName, 50, 67);

      describe(
        '別々の行に書かれた"animals"と"monsters"という単語。テキストは灰色の背景に黒で表示されています。'
      );
    }
