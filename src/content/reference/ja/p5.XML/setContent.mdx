    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のreptile子要素を取得します。
      let reptile = myXML.getChild('reptile');

      // reptileの元のコンテンツを取得します。
      let oldContent = reptile.getContent();

      // reptileのコンテンツを設定します。
      reptile.setContent('Loggerhead');

      // reptileの新しいコンテンツを取得します。
      let newContent = reptile.getContent();

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素の古いコンテンツと新しいコンテンツを表示します。
      text(`${oldContent}: ${newContent}`, 5, 50, 90);

      describe(
        `灰色の背景に緑色で "${oldContent}: ${newContent}" というテキストが書かれています。`
      );
    }
