  <p>要素が指定された属性を持っている場合は <code>true</code> を、持っていない場合は

  <p>注意：属性の値を取得するには

  <a href="/reference/p5.XML/getString/">myXML.getString()</a> または

  <a href="/reference/p5.XML/getNum/">myXML.getNum()</a> を使用してください。</p>

example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のmammal子要素を取得します。
      let mammal = myXML.getChild('mammal');

      // 要素がspecies属性を持っているかチェックします。
      let hasSpecies = mammal.hasAttribute('species');

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素がspecies属性を持っているかどうかを表示します。
      if (hasSpecies === true) {
        text('Species', 50, 50);
      } else {
        text('No species', 50, 50);
      }

      describe('灰色の背景に黒で "Species" と書かれたテキスト。');
    }
