description: >
  <p>属性の値を文字列として返します。</p>

  <p>最初のパラメータ <code>name</code> は、チェックする属性の名前を示す文字列です。

  例えば、<code>myXML.getString('color')</code> を呼び出すと、

  要素の <code>id</code> 属性が文字列として返されます。</p>

  <p>2番目のパラメータ <code>defaultValue</code> はオプションです。文字列が渡された場合、

  例えば <code>myXML.getString('color', 'deeppink')</code> のように、

  属性が存在しない場合にこの値が返されます。</p>

  <p>注意：属性の値を返すには、

  <a href="/reference/p5.XML/getString/">myXML.getString()</a> または

  <a href="/reference/p5.XML/getNum/">myXML.getNum()</a> を使用してください。</p>

    <code>
    let myXML;

    // XMLをロードしてp5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のreptile子要素を取得します。
      let reptile = myXML.getChild('reptile');

      // reptileのコンテンツを取得します。
      let content = reptile.getContent();

      // reptileの種を取得します。
      let species = reptile.getString('species');

      // テキストのスタイルを設定します。
      textAlign(LEFT, CENTER);
      textFont('Courier New');
      textSize(14);

      // species属性を表示します。
      text(`${content}: ${species}`, 5, 50, 90);

      describe(`灰色の背景に黒で "${content}: ${species}" と書かれたテキスト。`);
    }
