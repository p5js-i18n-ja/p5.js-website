    function preload() {

    function setup() {
      createCanvas(100, 100);

      background(200);

      // テキストのスタイルを設定します。

    // ユーザーがダブルクリックしたときにファイルを保存します。
    function doubleClicked() {
      // p5.PrintWriterオブジェクトを作成します。
      // ファイル形式として.xmlを使用します。
      let myWriter = createWriter('animals', 'xml');

      // XMLデータを文字列にシリアライズします。
      let data = myXML.serialize();

      // データをプリントストリームに書き込みます。
      myWriter.write(data);

      // ファイルを保存し、プリントストリームを閉じます。
      myWriter.close();
    }
