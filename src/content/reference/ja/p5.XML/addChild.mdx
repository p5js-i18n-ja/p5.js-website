description: >
  <p>新しい子要素を追加し、その参照を返します。</p>

  <p>パラメータ<code>child</code>は、子要素として追加する<a
  href="/reference/p5/p5.XML">p5.XML</a>オブジェクトです。

  例えば、<code>myXML.addChild(otherXML)</code>を呼び出すと、

  <code>otherXML</code>が<code>myXML</code>の子要素として挿入されます。</p>
line: 538
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLをロードしてp5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 新しいp5.XMLオブジェクトを作成します。
      let newAnimal = new p5.XML();

      // プロパティを設定します。
      newAnimal.setName('hydrozoa');
      newAnimal.setAttribute('id', 4);
      newAnimal.setAttribute('species', 'Physalia physalis');
      newAnimal.setContent('Bluebottle');

      // 子要素を追加します。
      myXML.addChild(newAnimal);

      // hydrozoaである最初の子要素を取得します。
      let blueBottle = myXML.getChild('hydrozoa');

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 子要素のコンテンツを取得します。
      let content = blueBottle.getContent();

      // 子要素のコンテンツを表示します。
      text(content, 50, 50);

      describe('灰色の背景に黒で「Bluebottle」という単語が書かれています。');
    }
    </code>
    </div>
class: p5.XML
