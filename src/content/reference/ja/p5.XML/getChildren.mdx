  <p>要素の子要素を新しい<a href="/reference/p5/p5.XML">p5.XML</a>オブジェクトとして含む配列を返します。</p>

  <p>パラメータ<code>name</code>はオプションです。<code>myXML.getChildren('cat')</code>のように文字列が渡された場合、
  メソッドは<code><cat></code>タグを持つ子要素のみを返します。</p>

itemtype: method
example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 子要素の配列を取得します。
      let children = myXML.getChildren();

      // テキストのスタイルを設定します。
      textAlign(LEFT, CENTER);
      textFont('Courier New');
      textSize(14);

      // 配列を反復処理します。
      for (let i = 0; i < children.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 20;

        // 子要素のコンテンツを取得します。
        let content = children[i].getContent();

        // 子要素のコンテンツを表示します。
        text(content, 10, y);
      }

      describe(
        '「Goat」、「Leopard」、「Zebra」、「Turtle」という単語が別々の行に書かれています。テキストは灰色の背景に黒で表示されています。'
      );
    }
    </code>
