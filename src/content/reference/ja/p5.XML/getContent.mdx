  <p>要素の内容を<code>String</code>として返します。</p>

  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のreptile子要素を取得します。
      let reptile = myXML.getChild('reptile');

      // reptileの内容を取得します。
      let content = reptile.getContent();

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素の内容を表示します。
      text(content, 5, 50, 90);

      describe(`灰色の背景に緑色で"${content}"というテキストが書かれています。`);
    }
    </code>
