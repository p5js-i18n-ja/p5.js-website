  <p>属性に指定された値を設定します。</p>

  <p>最初のパラメータ<code>name</code>は、設定する属性の名前を示す文字列です。</p>

  <p>2番目のパラメータ<code>value</code>は、属性の新しい値です。例えば、
  <code>myXML.setAttribute('id', 123)</code>を呼び出すと、<code>id</code>
  属性の値が123に設定されます。</p>

example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のreptile子要素を取得します。
      let reptile = myXML.getChild('reptile');

      // reptileの色を設定します。
      reptile.setAttribute('color', 'green');

      // reptileのコンテンツを取得します。
      let content = reptile.getContent();

      // reptileの色を取得します。
      let attribute = reptile.getString('color');

      // テキストのスタイルを設定します。
      textAlign(LEFT, CENTER);
      textFont('Courier New');
      textSize(14);

      // 要素のコンテンツを表示します。
      text(`${content} is ${attribute}`, 5, 50, 90);

      describe(
        `"${content} is ${attribute}"というテキストが灰色の背景に緑色で書かれています。`
      );
    }
    </code>
