description: >
  <p>属性の値を<code>Number</code>として返します。</p>

  <p>最初のパラメータ<code>name</code>は、チェックする属性の名前を示す文字列です。
  例えば、<code>myXML.getNum('id')</code>を呼び出すと、要素の<code>id</code>属性を
  数値として返します。</p>

  <p>2番目のパラメータ<code>defaultValue</code>はオプションです。数値が渡された場合、
  例えば<code>myXML.getNum('id', -1)</code>のように、属性が存在しないか
  数値に変換できない場合にこの値が返されます。</p>

  <p>注意：属性の値を返すには、

  <a href="/reference/p5.XML/getString/">myXML.getString()</a>または

  <a href="/reference/p5.XML/getNum/">myXML.getNum()</a>を使用してください。</p>

    <code>
    let myXML;

    // XMLをロードしてp5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 最初のreptile子要素を取得します。
      let reptile = myXML.getChild('reptile');

      // reptileのコンテンツを取得します。
      let content = reptile.getContent();

      // reptileのIDを取得します。
      let id = reptile.getNum('id');

      // テキストのスタイルを設定します。
      textAlign(LEFT, CENTER);
      textFont('Courier New');
      textSize(14);

      // ID属性を表示します。
      text(`${content} is ${id + 1}th`, 5, 50, 90);

      describe(`灰色の背景に黒で "${content} is ${id + 1}th" と書かれたテキスト。`);
    }
