description: >
  <p>最初に一致する子要素を新しい<a href="/reference/p5/p5.XML">p5.XML</a>オブジェクトとして返します。</p>

  <p>パラメータ<code>name</code>はオプションです。文字列が渡された場合（例：<code>myXML.getChild('cat')</code>）、
  タグ<code><cat></code>を持つ最初の子要素が返されます。数値が渡された場合（例：<code>myXML.getChild(1)</code>）、
  そのインデックスの子要素が返されます。</p>

example:
  - |-

    <div>
    <code>
    let myXML;

    // XMLを読み込み、p5.XMLオブジェクトを作成します。
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 哺乳類の最初の子要素を取得します。
      let goat = myXML.getChild('mammal');

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // 子要素のコンテンツを取得します。
      let content = goat.getContent();

      // 子要素のコンテンツを表示します。
      text(content, 50, 50);

      describe('灰色の背景に黒で「Goat」という単語が書かれています。');
    }
    </code>
    </div>
