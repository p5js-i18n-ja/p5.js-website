    function setup() {
      createCanvas(100, 100);

      background(200);

      // p5.Colorオブジェクトを作成します。
      let c = color(255, 128, 128);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 20, 35, 60);

      // アルファ値を変更します。
      c.setAlpha(128);

      // 右の長方形を描画します。
      fill(c);
      rect(50, 20, 35, 60);

      describe('2つの長方形。左側はサーモンピンク、右側は薄いピンク。');
