  <p>アニメーションGIFの現在のフレーム番号を取得します。</p>

  <p>返される値は0から始まります。つまり、最初のフレームは0、

  2番目のフレームは1、というようになります。</p>
line: 1960
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let gifAnimation;
    let currentFrame;

    // アニメーションGIFを読み込みます。
    function preload() {
      gifAnimation = loadImage('/assets/arnott-wallace-eye-loop-forever.gif');
    }

    function setup() {
      createCanvas(100, 100);
    }

    function draw() {
      background(200);

      // アニメーションGIFを表示します。
      image(gifAnimation, 0, 0);

      // 現在のフレーム番号を取得します。
      currentFrame = gifAnimation.getCurrentFrame();

      // フレーム番号を表示します。
      fill(0);
      text(`Frame: ${currentFrame}`, 10, 90);

      describe('まばたきする目のアニメーションGIF。現在のフレーム番号が下に表示されています。');
    }
    </code>
