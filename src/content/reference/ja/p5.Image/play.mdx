    <div>
    <code>
    let gif;
    let playing = true;

    function preload() {
      gif = loadImage('/assets/arnott-wallace-eye-loop-forever.gif');
    }

    function setup() {
      createCanvas(100, 100);
    }

    function draw() {
      background(255);

      // GIFを表示します。
      image(gif, 0, 0);

      // 再生状態を表示します。
      textAlign(CENTER);
      textSize(16);
      text(playing ? 'Playing' : 'Paused', 50, 90);

      describe('アニメーションGIF。クリックすると一時停止/再生します。');
    }

    // キャンバスをクリックすると一時停止/再生を切り替えます。
    function mousePressed() {
      if (playing) {
        gif.pause();
        playing = false;
      } else {
        gif.play();
        playing = true;
      }
    }
