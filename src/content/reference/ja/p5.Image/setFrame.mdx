---
title: setFrame
module: Image
submodule: Image
file: src/image/p5.Image.js
description: |
  <p>アニメーションGIFの現在のフレームを設定します。</p>
  <p>この関数は、アニメーションGIFの特定のフレームを表示します。
  フレーム番号は0から始まります。つまり、最初のフレームは0、
  2番目のフレームは1、というようになります。</p>
  <p>指定されたフレーム番号が有効な範囲外の場合、
  関数は何も行わず、現在のフレームは変更されません。</p>
line: 1590
params:
  - name: index
    description: |
      <p>表示するフレームのインデックス。</p>
    type: Number
itemtype: method
example:
  - |-

    <div>
    <code>
    let gif;
    let frameIndex = 0;

    function preload() {
      gif = loadImage('/assets/arnott-wallace-eye-loop-forever.gif');
    }

    function setup() {
      createCanvas(100, 100);
    }

    function draw() {
      background(255);

      // 現在のフレームを表示します。
      gif.setFrame(frameIndex);
      image(gif, 0, 0);

      // フレーム番号を表示します。
      textAlign(CENTER);
      textSize(16);
      text('Frame: ' + frameIndex, 50, 90);

      describe('アニメーションGIF。クリックすると次のフレームに進みます。');
    }

    // キャンバスをクリックすると次のフレームに進みます。
    function mousePressed() {
      frameIndex = (frameIndex + 1) % gif.numFrames();
    }
    </code>
    </div>
