  <p>アニメーションGIFのフレーム間の遅延時間を変更します。</p>

    let gifAnimation;

    // アニメーションGIFを読み込みます。
    function preload() {
      gifAnimation = loadImage('/assets/arnott-wallace-wink-loop-once.gif');
    }

    function setup() {
      createCanvas(100, 100);
      background(200);

      // アニメーションGIFを表示します。
      image(gifAnimation, 0, 0);

      // すべてのフレームの遅延時間を1000ミリ秒に設定します。
      gifAnimation.delay(1000);
