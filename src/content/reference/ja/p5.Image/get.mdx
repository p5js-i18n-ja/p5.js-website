description: >
  <p>画像の一部または全体のピクセルデータを取得します。</p>

  <p>引数なしで呼び出すと、画像全体のピクセルデータを含む

  <a href="/reference/p5/p5.Image">p5.Image</a> オブジェクトを返します。</p>

  <p>2つの数値引数 <code>x</code> と <code>y</code> を指定すると、その座標にある

  単一ピクセルの色を配列 <code>[R, G, B, A]</code> として返します。</p>

  <p>4つの数値引数 <code>x</code>、<code>y</code>、<code>w</code>、<code>h</code> を指定すると、

  指定された矩形領域のピクセルデータを含む <a href="/reference/p5/p5.Image">p5.Image</a>

  オブジェクトを返します。</p>

    let img;

    function setup() {
      createCanvas(100, 100);

      // 画像を表示します。
      image(img, 0, 0);

      // 画像の一部を取得します。
      let cropped = img.get(0, 0, 50, 50);

      // 切り取った画像を表示します。
      image(cropped, 50, 50);

    }
