description: >
  <p><code>String</code>を分割し、分割された部分を含む配列を返します。</p>

  <p>最初のパラメータ<code>value</code>は、分割する文字列です。</p>

  <p>2番目のパラメータ<code>delim</code>は、文字列を分割するために使用される文字（複数可）です。

  例えば、<code>split('rock...paper...scissors', '...')</code>を呼び出すと、

  <code>['rock', 'paper', 'scissors']</code>という配列が返されます。これは、各単語の間に

  3つのピリオド<code>...</code>があるためです。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // 文字列変数を作成します。
      let string = 'rock...paper...scissors';

      // 文字列を ... で分割します
      let words = split(string, '...');

      // 配列をコンソールに表示します:
      // ["rock", "paper", "scissors"]
      print(words);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(16);

      // words配列を反復処理します。
      for (let i = 0; i < words.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 25;

        // 単語を表示します。
        text(words[i], 50, y);
      }

      describe(
        '灰色の背景に黒字で、"rock"、"paper"、"scissors"という単語が別々の行に書かれています。'
      );
    }
