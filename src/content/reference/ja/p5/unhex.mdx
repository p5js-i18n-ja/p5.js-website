description: >
  <p>16進数の値を持つ<code>String</code>を<code>Number</code>に変換します。</p>

  <p><code>unhex()</code>は16進数の値を持つ文字列を数値に変換します。

  16進数（hex）は16進数で、16個のユニークな数字があることを意味します。16進数は0〜9の数字をA〜Fの文字で拡張します。例えば、

  10進数の<code>11</code>（11）は16進数では文字<code>B</code>と表記されます。</p>

  <p>最初のパラメータ<code>n</code>は変換する16進数の文字列です。例えば、

  <code>unhex('FF')</code>は数値255を返します。配列が渡された場合、

  <code>unhex(['00', '80', 'FF'])</code>のように、数値の配列が返されます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 数値の配列を作成
      let original = ['00', '80', 'FF'];

      // 数値を16進数の文字列に変換
      // 2桁の16進数のみを使用
      let converted = unhex(original, 2);

      // テキストのスタイルを設定
      textAlign(RIGHT, CENTER);
      textSize(16);

      // 変換された値を繰り返し処理
      for (let i = 0; i < converted.length; i += 1) {

        // y座標を計算
        let y = (i + 1) * 25;

        // 元の値と変換後の値を表示
        text(`${ original[i]} = ${converted[i]}`, 80, y);
      }

      describe(
        '3行に分けて書かれた "00 = 0"、"80 = 128"、"FF = 255" のテキスト。テキストはグレーの背景に黒で書かれています。'
      );
    }
