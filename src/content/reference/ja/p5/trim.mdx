description: >
  <p><code>String</code>の先頭と末尾から空白を削除し、中間部分は変更しません。</p>

  <p><code>trim()</code>は

  <a href="https://developer.mozilla.org/en-US/docs/Glossary/whitespace/"
  target="_blank">空白文字</a>

  （スペース、キャリッジリターン、タブ、Unicodeの"nbsp"文字など）を削除します。</p>

  <p>パラメータ<code>str</code>は、トリムする文字列です。単一の文字列が渡された場合

  （<code>trim('   pad   ')</code>など）、単一の文字列が返されます。文字列の配列が

  渡された場合（<code>trim(['    pad   ', '\n space \n'])</code>など）、文字列の

  配列が返されます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 文字列変数を作成します。
      let string = '   p5*js   ';

      // 空白をトリムします。
      let trimmed = trim(string);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);

      createCanvas(100, 100);

      background(200);

      // 文字列の配列を作成します。
      let strings = ['   wide  ', '\n  open  ', '\n spaces  '];

      // 空白をトリムします。
      let trimmed = trim(strings);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(10);

      // テキストを表示します。
      text(`${trimmed[0]} ${trimmed[1]} ${trimmed[2]}`, 50, 50);

    }
