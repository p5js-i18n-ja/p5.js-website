description: >
  <p>画像から環境光を作成します。</p>

  <p><code>imageLight()</code>はあらゆる方向から光が当たっているようにシミュレートします。
  その効果は

  スケッチを巨大な球体の中心に置き、その球体がその画像を

  テクスチャとして使用しているようなものです。画像の拡散光は

  <a href="/reference/p5/fill/">fill()</a>の影響を受け、鏡面反射は

  <a href="/reference/p5/specularMaterial/">specularMaterial()</a>と

  <a href="/reference/p5/shininess/">shininess()</a>の影響を受けます。</p>

  <p>パラメータ<code>img</code>は、光源として使用する
  <a href="/reference/p5/p5.Image">p5.Image</a>オブジェクトです。</p>

example:
  - |-

    <div class="notest">
    <code>
    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。

    let img;

    // 画像を読み込み、p5.Imageオブジェクトを作成します。
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('風景の上に浮かぶ球体。球体の表面には風景が反射しています。');
    }

    function draw() {
      // マウスでの軌道制御を有効にします。
      orbitControl();

      // 画像をパノラマ（360度背景）として描画します。
      panorama(img);

      // ソフトな環境光を追加します。
      ambientLight(50);

      // 画像から光を追加します。
      imageLight(img);

      // 球体のスタイルを設定します。
      specularMaterial(20);
      shininess(100);
      noStroke();

      // 球体を描画します。
      sphere(30);
    }
    </code>
