  <p>図形の塗りつぶし色の設定を無効にします。</p>

  <p><code>noFill()</code>を呼び出すことは、塗りつぶしを完全に透明にすることと同じです。

  <code>fill(0, 0)</code>のようになります。<a
  href="/reference/p5/noStroke/">noStroke()</a>と

  <code>noFill()</code>の両方が呼び出された場合、画面には何も描画されません。</p>

  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // 上の正方形を描画します。
      square(32, 10, 35);

      // 下の正方形を描画します。
      noFill();
      square(32, 55, 35);

      describe('黒い輪郭線を持つ白い正方形の上に、空の正方形があります。');
    }
    </code>
    </div>

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('黒いキャンバス上で回転する紫色の立方体のワイヤーフレーム。');
    }

    function draw() {
      background(0);

      // 立方体のスタイルを設定します。
      noFill();
      stroke(100, 100, 240);

      // 座標を回転させます。
      rotateX(frameCount * 0.01);
      rotateY(frameCount * 0.01);

      // 立方体を描画します。
      box(45);
    }
