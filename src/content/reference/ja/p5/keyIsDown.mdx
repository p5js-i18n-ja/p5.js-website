description: >
  <p>チェックしているキーが押されている場合は<code>true</code>を返し、
  押されていない場合は<code>false</code>を返します。</p>

  <p><code>keyIsDown()</code>は、複数の異なるキー押下を
  チェックする際に便利です。

  例えば、<code>keyIsDown()</code>を使用して<code>LEFT_ARROW</code>と

  <code>UP_ARROW</code>の両方が押されているかどうかを確認できます：</p>

  <pre><code class="language-js">if (keyIsDown(LEFT_ARROW) &&
  keyIsDown(UP_ARROW)) {
    // 斜めに移動する。
  }

  </code></pre>

  <p><code>keyIsDown()</code>は、<code>keyIsDown(37)</code>や

  <code>keyIsDown(LEFT_ARROW)</code>のように、

  <a href="/reference/p5/keyCode/">keyCode</a>の値を使用してキー押下を
  チェックできます。キーコードは

  <a href="https://keycode.info" target="_blank">keycode.info</a>などの
  ウェブサイトで確認できます。</p>

    // キャンバスをクリックしてキー押下の検出を開始します。

    let x = 50;
    let y = 50;

    function setup() {
      createCanvas(100, 100);

      background(200);

      describe(
        'グレーの正方形の中央に黒い円があります。ユーザーが矢印キーを押すと円が移動します。移動すると軌跡が残ります。'
      );
    }

    function draw() {
      // 矢印キーが押されたらxとyを更新します。
      if (keyIsDown(LEFT_ARROW) === true) {
        x -= 1;
      }

      if (keyIsDown(RIGHT_ARROW) === true) {
        x += 1;
      }

      if (keyIsDown(UP_ARROW) === true) {
        y -= 1;
      }

      if (keyIsDown(DOWN_ARROW) === true) {
        y += 1;
      }

      // 円のスタイルを設定します。
      fill(0);

      // 円を描画します。
      circle(x, y, 5);
    }
