description: >
  <p><a href="/reference/p5/pointLight/">pointLight()</a>と<a
  href="/reference/p5/spotLight/">spotLight()</a>の減衰率を設定します。</p>

  <p>光の減衰は、距離に応じてビームの強度がどのように変化するかを表します。

  例えば、ランタンはゆっくりとした減衰、懐中電灯は中程度の減衰、

  レーザーポインターは急激な減衰を持ちます。</p>

  <p><code>lightFalloff()</code>には3つのパラメータ、<code>constant</code>、<code>linear</code>、

  <code>quadratic</code>があります。これらは距離<code>d</code>における減衰を

  計算するための数値で、以下のように使用されます：</p>

  <p><code>falloff = 1 / (constant + d * linear + (d * d) * quadratic)</code></p>

  <p>注意：<code>constant</code>、<code>linear</code>、<code>quadratic</code>は常に

  0より大きい値に設定する必要があります。</p>

    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。
    // ダブルクリックすると減衰率が変更されます。

    let useFalloff = false;

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('灰色の背景に描かれた球体。ユーザーがダブルクリックすると光の強度が変化します。');
    }

    function draw() {
      background(50);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // 光の減衰を設定します。
      if (useFalloff === true) {
        lightFalloff(2, 0, 0);
      }

      // 前方から白い点光源を追加します。
      pointLight(255, 255, 255, 0, 0, 100);

      // 球体のスタイルを設定します。
      noStroke();

      // 球体を描画します。
      sphere(30);
    }

    // ユーザーがダブルクリックしたときに減衰値を変更します。
    function doubleClicked() {
      useFalloff = true;
    }
