description: >
  <p>2つの色を混ぜ合わせて、その間の第3の色を見つけます。</p>

  <p><code>amt</code>パラメータは、2つの値の間を補間する量を指定します。

  0は最初の色に等しく、0.1は最初の色に非常に近く、

  0.5は2つの色の中間点であり、以下同様です。負の数は0に設定されます。

  1より大きい数は1に設定されます。これは<a href="/reference/lerp/">lerp</a>の

  動作とは異なります。[0, 1]の区間外の数値は奇妙で予期しない色を

  生成するため、これは必要です。</p>

  <p>色の補間方法は、現在の<a href="/reference/colorMode/">colorMode()</a>に依存します。</p>

  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // 補間する2つのp5.Colorオブジェクトを作成します。
      let from = color(218, 165, 32);
      let to = color(72, 61, 139);

      // 中間色を作成します。
      let interA = lerpColor(from, to, 0.33);
      let interB = lerpColor(from, to, 0.66);

      // 左の長方形を描画します。
      noStroke();
      fill(from);
      rect(10, 20, 20, 60);

      // 左中央の長方形を描画します。
      fill(interA);
      rect(30, 20, 20, 60);

      // 右中央の長方形を描画します。
      fill(interB);
      rect(50, 20, 20, 60);

      // 右の長方形を描画します。
      fill(to);
      rect(70, 20, 20, 60);

      describe(
        '4つの長方形。左から右へ、長方形はタン、茶色、茶紫色、紫色です。'
      );
    }
    </code>
