    <div class="norender">
    <code>
    // このサンプルをモバイルデバイスで実行してください
    // 四角形の色が白に近づくにつれて、値を変更するには
    // デバイスをより大きく動かす必要があります

    let value = 0;
    let threshold = 0.5;
    function setup() {
      setMoveThreshold(threshold);
    }
    function draw() {
      fill(value);
      rect(25, 25, 50, 50);
      describe(`キャンバスの中央に50x50の黒い四角形。
        モバイルでデバイスを動かすと白くなる`);
    }
    function deviceMoved() {
      value = value + 5;
      threshold = threshold + 0.1;
      if (value > 255) {
        value = 0;
        threshold = 30;
      }
      setMoveThreshold(threshold);
    }
