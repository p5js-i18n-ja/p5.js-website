    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // ボタンを作成し、キャンバスの下に配置します。
      let button = createButton('click me');
      button.position(0, 100);

      // ボタンが押されたときにrepaint()を呼び出します。

    <div>
    <code>
    let button;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // ボタンを作成し、値を0に設定します。
      // ボタンをキャンバスの下に配置します。
      button = createButton('click me', 'red');
      button.position(0, 100);

      // ボタンが押されたときにrandomColor()を呼び出します。
      button.mousePressed(randomColor);

      describe('「click me」と書かれたボタンがある赤い正方形。ボタンをクリックすると正方形の色が変わります。');
    }

    function draw() {
      // ボタンの値を使用して背景色を設定します。
      let c = button.value();
      background(c);
    }

    // ボタンの値をランダムな色に設定します。
    function randomColor() {
      let c = random(['red', 'green', 'blue', 'yellow']);
      button.value(c);
    }
    </code>
