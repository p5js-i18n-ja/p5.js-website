    <div class="norender">
    <code>
    // このサンプルをモバイルデバイスで実行してください
    // ボックスの塗りつぶしが白に近づくほど、値を変更するには
    // デバイスをより強く振る必要があります。

    let value = 0;
    let threshold = 30;
    function setup() {
      setShakeThreshold(threshold);
    }
    function draw() {
      fill(value);
      rect(25, 25, 50, 50);
      describe(`キャンバスの中央に50x50の黒い四角形。
        モバイルでデバイスを振ると白くなります`);
    }
    function deviceMoved() {
      value = value + 5;
      threshold = threshold + 5;
      if (value > 255) {
        value = 0;
        threshold = 30;
      }
      setShakeThreshold(threshold);
    }
