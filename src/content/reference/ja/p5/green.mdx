description: >
  <p>色の緑の値を取得します。</p>

  <p><code>green()</code>は<a href="/reference/p5/p5.Color">p5.Color</a>オブジェクト、色成分の配列、

  またはCSSの色文字列から緑の値を抽出します。</p>

  <p>デフォルトでは、<code>green()</code>は色の緑の値を0から255の範囲で返します。

  <a href="/reference/p5/colorMode/">colorMode()</a>がRGBに設定されている場合、

  指定された範囲内で緑の値を返します。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // p5.Colorオブジェクトを作成します。
      let c = color(175, 100, 220);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // 'greenValue'を100に設定します。
      let greenValue = green(c);

      // 右の長方形を描画します。
      fill(0, greenValue, 0);
      rect(50, 15, 35, 70);

      describe('2つの長方形。左側は薄紫色で、右側は濃い緑色です。');
    }
