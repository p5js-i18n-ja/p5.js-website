description: >
  <p><code>Number</code>または<code>String</code>を1文字の<code>String</code>に変換します。</p>

  <p><code>char()</code>は数値を対応する1文字の文字列表現に変換します。</p>

  <p>パラメータ<code>n</code>は変換する値です。数値が渡された場合、例えば

  <code>char(65)</code>のように、対応する1文字の文字列が返されます。

  文字列が渡された場合、例えば<code>char('65')</code>のように、文字列は整数

  （整数）に変換され、対応する1文字の文字列が返されます。

  配列が渡された場合、例えば<code>char([65, 66, 67])</code>のように、

  1文字の文字列の配列が返されます。</p>

  <p>変換についての詳細は<a
  href="https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String/fromCharCode/"
  target="_blank">MDN</a>

  を参照してください。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 数値の配列を作成します。
      let original = ['65', 66, '67'];

      // 文字列をcharに変換します。
      let converted = char(original);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textSize(16);

      // 変換された配列の要素を反復処理します。
      for (let i = 0; i < converted.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 25;

        // 元の値と変換後の値を表示します。
        text(`${original[i]} : ${converted[i]}`, 50, y);
      }

      describe(
        '3行に分けて書かれた "65 : A"、"66 : B"、"67 : C" というテキスト。テキストはグレーの背景に黒で書かれています。'
      );
    }
    </code>
