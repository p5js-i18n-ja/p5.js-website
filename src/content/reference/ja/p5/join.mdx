  <p>文字列の配列を1つの文字列に結合します。</p>

  <p>最初のパラメータ<code>list</code>は、結合する文字列の配列です。</p>

  <p>2番目のパラメータ<code>separator</code>は、結合された文字列を区切るために使用される文字（複数可）です。

  例えば、<code>join(myWords, ' : ')</code>を呼び出すと、各単語がコロンとスペースで

  区切られた文字列が返されます。</p>

      background(200);

      // 文字列の配列を作成します。
      let myWords = ['one', 'two', 'three'];

      // 結合された文字列を作成します
      let combined = join(myWords, ' : ');

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);

      // 結合された文字列を表示します。
      text(combined, 50, 50);

      describe('グレーのキャンバスに黒で書かれた "one : two : three" というテキスト。');
    }
