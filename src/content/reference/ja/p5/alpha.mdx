  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // p5.Colorオブジェクトを作成します。
      let c = color(0, 126, 255, 102);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // 'alphaValue'を102に設定します。
      let alphaValue = alpha(c);

      // 右の長方形を描画します。
      fill(alphaValue);
      rect(50, 15, 35, 70);

      describe('2つの長方形。左側は薄い青色で、右側は炭色です。');
    }
    </code>
