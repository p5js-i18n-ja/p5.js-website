  <p>シーンに環境光と指向性ライトを配置します。

  draw() 内で（直接的または間接的に）呼び出す必要があります。
  setup() 内に配置すると、ループの最初の1回だけ効果があります。</p>

    <code>
    // クリックしてドラッグすると、異なる角度からシーンを見ることができます。
    // ダブルクリックするとライトがオンになります。

    let isLit = false;

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('グレーの背景に描かれた白い箱。ユーザーがダブルクリックすると、光の質が変化します。');
    }

    function draw() {
      background(50);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // ライトを制御します。
      if (isLit === true) {
        lights();
      }

      // 箱を描画します。
      box();
    }

    // ユーザーがダブルクリックするとライトがオンになります。
    function doubleClicked() {
      isLit = true;
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('グレーの背景に描かれた白い箱。');
    }

    function draw() {
      background(50);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // ライトをオンにします。
      ambientLight(128, 128, 128);
      directionalLight(128, 128, 128, 0, 0, -1);

      // 箱を描画します。
      box();
    }
