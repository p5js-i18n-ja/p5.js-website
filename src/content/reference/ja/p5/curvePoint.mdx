description: >
  <p>補間を使用してスプライン曲線に沿った座標を計算します。</p>

  <p><code>curvePoint()</code>はアンカーポイントとコントロールポイントを使用して、
  スプライン曲線に沿った座標を計算します。これは<a href="/reference/p5/curve/">curve()</a>
  関数と同じ順序でポイントを期待します。<code>curvePoint()</code>は一度に1つの軸に対して
  機能します。アンカーポイントとコントロールポイントのx座標を渡すと、曲線上の点の
  x座標を計算します。アンカーポイントとコントロールポイントのy座標を渡すと、
  曲線上の点のy座標を計算します。</p>

  <p>最初のパラメータ<code>a</code>は、最初のコントロールポイントの座標です。</p>

  <p>2番目と3番目のパラメータ<code>b</code>と<code>c</code>は、アンカーポイントの座標です。</p>

  <p>4番目のパラメータ<code>d</code>は、最後のコントロールポイントの座標です。</p>

  <p>5番目のパラメータ<code>t</code>は、曲線に沿って補間する量です。0は最初の
  アンカーポイント、1は2番目のアンカーポイント、0.5はその中間点です。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      let x1 = 5;

      let y4 = 65;

      // 曲線を描画します。
      noFill();
      curve(x1, y1, x2, y2, x3, y3, x4, y4);

      // 曲線のパスに沿って円を描画します。
      fill(255);

      // 上部
      let x = curvePoint(x1, x2, x3, x4, 0);
      let y = curvePoint(y1, y2, y3, y4, 0);
      circle(x, y, 5);

      // 中央
      x = curvePoint(x1, x2, x3, x4, 0.5);
      y = curvePoint(y1, y2, y3, y4, 0.5);
      circle(x, y, 5);

      // 下部
      x = curvePoint(x1, x2, x3, x4, 1);
      y = curvePoint(y1, y2, y3, y4, 1);
      circle(x, y, 5);

      describe('灰色の正方形上の黒い曲線。曲線の端点と中心が白い円で示されています。');
    }

    function setup() {
      createCanvas(100, 100);

      describe('灰色の正方形上の黒い曲線。白い円が曲線に沿って前後に動きます。');
    }

    function draw() {
      background(200);

      // 曲線のアンカーポイントとコントロールポイントの座標を設定します。
      let x1 = 5;
      let y1 = 26;
      let x2 = 73;
      let y2 = 24;
      let x3 = 73;
      let y3 = 61;
      let x4 = 15;
      let y4 = 65;

      // 曲線を描画します。
      noFill();
      curve(x1, y1, x2, y2, x3, y3, x4, y4);

      // 円の座標を計算します。
      let t = 0.5 * sin(frameCount * 0.01) + 0.5;
      let x = curvePoint(x1, x2, x3, x4, t);
      let y = curvePoint(y1, y2, y3, y4, t);

      // 円を描画します。
      fill(255);
      circle(x, y, 5);
    }
