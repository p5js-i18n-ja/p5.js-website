description: >
  <p>画像に色合いを適用します。</p>

  <p><code>tint()</code>は、後続の全ての<code>image()</code>呼び出しに影響します。
  画像を元の色で表示するには、<a href="/reference/p5/noTint">noTint()</a>を使用します。</p>

  <p>パラメータは<a href="/reference/p5/fill">fill()</a>と同じように指定します。</p>

  <p>画像の透明度を変更するには、オプションのアルファパラメータを使用するか、
  <code>tint()</code>の代わりに<a href="/reference/p5/image">image()</a>の
  オプションのアルファパラメータを使用します。</p>

    <div>
    <code>
    let img;

    // 画像を読み込みます。
    function preload() {
      img = loadImage('/assets/laDefense.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      image(img, 0, 0);
      tint(0, 153, 204);  // 青色の色合い
      image(img, 50, 0);

      describe('2つの画像。左側は元の画像で、右側は青色の色合いがかかっています。');
    }
