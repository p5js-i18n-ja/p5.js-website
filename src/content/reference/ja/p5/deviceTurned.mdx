  <p><a href="/reference/p5/deviceTurned/">deviceTurned()</a>関数は、デバイスが

  90度以上連続して回転したときに呼び出されます。</p>

  <p><a href="/reference/p5/deviceTurned/">deviceTurned()</a>メソッドをトリガーする軸は

  turnAxis変数に格納されます。<a href="/reference/p5/deviceTurned/">deviceTurned()</a>メソッドは

  X、Y、Z軸のいずれかでトリガーするようにロックできます。

  turnAxis変数を'X'、'Y'、'Z'と比較することで設定できます。</p>

    // このサンプルをモバイルデバイスで実行してください
    // デバイスを90度回転させて
    // 値を変更します。

    let value = 0;
    function draw() {
      fill(value);
      rect(25, 25, 50, 50);
      describe(`キャンバスの中央に50x50の黒い四角形。
        モバイルでデバイスを回転させると白に変わります`);
    }
    function deviceTurned() {

    // このサンプルをモバイルデバイスで実行してください

    let value = 0;

    function deviceTurned() {
      if (turnAxis === 'X') {

          value = 0;

      }
