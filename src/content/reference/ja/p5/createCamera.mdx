description: >
  <p>新しい<a href="/reference/p5/p5.Camera">p5.Camera</a>オブジェクトを作成し、

  現在の（アクティブな）カメラとして設定します。</p>

  <p>新しいカメラは、デフォルトの位置<code>(0, 0, 800)</code>とデフォルトの

  透視投影で初期化されます。そのプロパティは

  <a href="/reference/p5/p5.Camera">p5.Camera</a>メソッド（例：

  <code>myCamera.lookAt(0, 0, 0)</code>）で制御できます。</p>

  <p>注意：すべての3Dスケッチは、初期化されたデフォルトカメラで開始します。

  このカメラは、シーン内で唯一のカメラである場合、

  <a href="/reference/p5/camera/">camera()</a>、

  <a href="/reference/p5/perspective/">perspective()</a>、

  <a href="/reference/p5/ortho/">ortho()</a>、

  <a href="/reference/p5/frustum/">frustum()</a>関数で制御できます。</p>

  <p>注意：<code>createCamera()</code>はWebGLモードでのみ使用できます。</p>

    // ダブルクリックでカメラを切り替えます。

    let cam1;
    let cam2;
    let usingCam1 = true;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 最初のカメラを作成します。
      // デフォルト設定のままにします。
      cam1 = createCamera();

      // 2番目のカメラを作成します。
      // 左上に配置します。
      // 原点を向くようにします。
      cam2 = createCamera();
      cam2.setPosition(400, -400, 800);
      cam2.lookAt(0, 0, 0);

      // 現在のカメラをcam1に設定します。
      setCamera(cam1);

      describe('灰色の背景に白い立方体。ユーザーがダブルクリックすると、カメラが正面図と俯瞰図の間で切り替わります。');
    }

    function draw() {
      background(200);

      // 箱を描画します。
      box();
    }

    // ユーザーがダブルクリックしたときに現在のカメラを切り替えます。
    function doubleClicked() {
      if (usingCam1 === true) {
        setCamera(cam2);
        usingCam1 = false;
      } else {
        setCamera(cam1);
        usingCam1 = true;
      }
    }
