description: >
  <p><code>String</code>を分割し、分割された部分を含む配列を返します。</p>

  <p><code>splitTokens()</code>は

  <a href="/reference/p5/split/">split()</a>の拡張版です。リストから任意の文字が

  検出されたときに文字列を分割できます。</p>

  <p>最初のパラメータ<code>value</code>は、分割する文字列です。</p>

  <p>2番目のパラメータ<code>delim</code>はオプションです。文字列を分割するために

  使用する文字を設定します。<code>delim</code>は単一の文字列（例：

  <code>splitTokens('rock...paper...scissors...shoot', '...')</code>）または

  文字列の配列（例：

  <code>splitTokens('rock;paper,scissors...shoot, [';', ',', '...'])</code>）にすることができます。

  デフォルトでは、<code>delim</code>文字が指定されていない場合、

  任意の空白文字が分割に使用されます。空白文字にはタブ（<code>\t</code>）、改行（<code>\n</code>）、

  キャリッジリターン（<code>\r</code>）、フォームフィード（<code>\f</code>）、スペースが含まれます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 文字列変数を作成します。
      let string = 'rock paper scissors shoot';

      // 各スペースで文字列を分割します。
      let words = splitTokens(string);

      // 配列をコンソールに出力します。
      print(words);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(12);

      // words配列を反復処理します。
      for (let i = 0; i < words.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 20;

        // 単語を表示します。
        text(words[i], 50, y);
      }

      );
