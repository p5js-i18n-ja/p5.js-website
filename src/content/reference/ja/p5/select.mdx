description: >
  <p>ページ内で、指定された<a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">CSSセレクタ文字列</a>に一致する最初の要素を検索します。</p>

  <p>セレクタ文字列には、ID、クラス、タグ名、またはそれらの組み合わせを使用できます。

  <code>select()</code>は、一致する要素が見つかった場合は<a
  href="/reference/p5/p5.Element">p5.Element</a>オブジェクトを返し、

  見つからなかった場合は<code>null</code>を返します。</p>

  <p>2番目のパラメータ<code>container</code>はオプションです。これは検索対象となる
  
  コンテナを指定します。<code>container</code>には、CSSセレクタ文字列、

  <a href="/reference/p5/p5.Element">p5.Element</a>オブジェクト、または

  <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement/"
  target="_blank">HTMLElement</a>オブジェクトを指定できます。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);
      background(200);

      // キャンバスをタグで選択します。
      let cnv = select('canvas');
      cnv.style('border', '5px deeppink dashed');

      describe('点線のピンクの枠線がある灰色の正方形。');
    }

    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // キャンバスにクラス属性を追加します。
      cnv.class('pinkborder');

      background(200);

      // キャンバスをクラスで選択します。
      cnv = select('.pinkborder');

      // 枠線のスタイルを設定します。
      cnv.style('border', '5px deeppink dashed');

      describe('点線のピンクの枠線がある灰色の正方形。');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // キャンバスにIDを設定します。
      cnv.id('mycanvas');

      background(200);

      // キャンバスをIDで選択します。
      cnv = select('#mycanvas');

      // 枠線のスタイルを設定します。
      cnv.style('border', '5px deeppink dashed');

      describe('点線のピンクの枠線がある灰色の正方形。');
    }
    </code>
