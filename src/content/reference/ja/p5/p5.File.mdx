    // ファイル入力を使用して、ロードして表示する
    // 画像を選択します。
    let img;

    function setup() {
      createCanvas(100, 100);

      // ファイル入力を作成し、キャンバスの下に配置します。
      // 画像ファイルがロードされたらhandleImage()を呼び出します。
      let input = createFileInput(handleImage);
      input.position(0, 100);

      describe('グレーの正方形と、その下にファイル入力があります。ユーザーがロードする画像ファイルを選択すると、その画像が正方形に表示されます。');
    }

    function draw() {
      background(200);

      // 画像が準備できたら描画します。
      if (img) {
        image(img, 0, 0, width, height);
      }
    }

    // p5.Fileのデータがロードされたら使用します。
    function handleImage(file) {
      // p5.Fileのタイプをチェックします。
      if (file.type === 'image') {
        // p5.Fileのデータを使用して画像を作成します。
        img = createImg(file.data, '');

        // 画像要素を非表示にして、2回表示されないようにします。
        img.hide();
      } else {
        img = null;
      }
    }
