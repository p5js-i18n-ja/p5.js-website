description: >
  <p>色の彩度の値を取得します。</p>

  <p><code>saturation()</code>は<a href="/reference/p5/p5.Color">p5.Color</a>オブジェクト、色成分の配列、

  またはCSSの色文字列から彩度の値を抽出します。</p>

  <p>彩度はHSBとHSLで異なる方法でスケーリングされます。デフォルトでは、<code>saturation()</code>は

  色のHSLの彩度を0から100の範囲で返します。<a href="/reference/colorMode/">colorMode()</a>が

  HSBまたはHSLに設定されている場合、指定されたモードでの彩度の値を返します。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(50);

      // HSB色を使用します。
      colorMode(HSB);

      // p5.Colorオブジェクトを作成します。
      let c = color(0, 50, 100);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // 'satValue'を50に設定します。
      let satValue = saturation(c);

      // 右の長方形を描画します。
      fill(satValue);
      rect(50, 15, 35, 70);

      describe('2つの長方形。左側はサーモンピンクで、右側は濃い灰色です。');
    }
