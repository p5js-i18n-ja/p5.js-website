  <p>没入型の3D背景を作成します。</p>

  <p><code>panorama()</code>は、地図やHDRIなどの360˚コンテンツを含む画像を、
  スケッチを取り囲む没入型の3D背景に変換します。空間を探索するには、
  <a href="/reference/p5/orbitControl/">orbitControl()</a>や
  <a href="/reference/p5/camera/">camera()</a>などの関数を使用して
  カメラの視点を変更する必要があります。</p>

example:
  - |-

    <div class="notest">
    <code>
    // マウスをクリックしてドラッグすると、異なる角度からシーンを見ることができます。

    let img;

    // 画像を読み込み、p5.Imageオブジェクトを作成します。
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100 ,100 ,WEBGL);

      describe('風景の上に浮かぶ球体。球体の表面には風景が反射しています。ユーザーがマウスをドラッグすると、完全な風景を3Dで見ることができます。');
    }

    function draw() {
      // パノラマを追加します。
      panorama(img);

      // マウスで軌道制御を有効にします。
      orbitControl();

      // 画像を光源として使用します。
      imageLight(img);

      // 球体のスタイルを設定します。
      noStroke();
      specularMaterial(50);
      shininess(200);
      metalness(100);

      // 球体を描画します。
      sphere(30);
    }
    </code>
    </div>
