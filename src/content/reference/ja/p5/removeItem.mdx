description: >
  <p>ウェブブラウザのローカルストレージからアイテムを削除します。</p>

  <p>ウェブブラウザは、組み込みの

  <a
  href="https://developer.mozilla.org/en-US/docs/Web/API/Window/localStorage/"
  target="_blank">localStorage オブジェクト</a>を使用して

  少量のデータを保存できます。<code>localStorage</code>に保存されたデータは、

  ページの更新やブラウザの再起動後でも、いつでも取得できます。データは

  キーと値のペアとして保存されます。</p>

  <p><a href="/reference/p5/storeItem/">storeItem()</a>を使用すると、

  <code>localStorage</code>に値を簡単に保存でき、<code>removeItem()</code>を使用すると

  簡単に削除できます。</p>

  <p>パラメータ<code>key</code>は、削除する値の名前を文字列で指定します。

  例えば、<code>removeItem('size')</code>を呼び出すと、キーが<code>size</code>の

  アイテムが削除されます。</p>

  <p>注意：パスワードや個人情報などの機密データは、<code>localStorage</code>に

  保存しないでください。</p>

    <div>
    <code>
    // ダブルクリックしてlocalStorageからアイテムを削除します。

    function setup() {
      createCanvas(100, 100);

      // プレイヤーの名前を保存します。
      storeItem('name', 'Feist');

      // プレイヤーのスコアを保存します。
      storeItem('score', 1234);

      describe(
        'グレーの背景に黒で "Feist: 1234" と書かれたテキスト。ユーザーがダブルクリックすると "Feist: null" と表示されます。'
      );
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textSize(14);

      // 名前を取得します。
      let name = getItem('name');

      // スコアを取得します。
      let score = getItem('score');

      // スコアを表示します。
      text(`${name}: ${score}`, 50, 50);
    }

    // ユーザーがダブルクリックしたときにlocalStorageから単語を削除します。
    function doubleClicked() {
      removeItem('score');
    }
