  <p>p5.jsによって作成されたすべての要素を、イベントハンドラーを含めて削除します。</p>

  <a href="/reference/p5/createCanvas/">createCanvas()</a>によって作成されたcanvas要素

  および<a href="/reference/p5/createGraphics/">createGraphics()</a>によって作成された

  <a href="/reference/p5/p5.Renderer">p5.Render</a>オブジェクト。</p>

      createCanvas(100, 100);

      background(200);

      // 段落要素を作成し、
      // キャンバスの中央に配置します。
      let p = createP('p5*js');
      p.position(25, 25);

      describe('中央に"p5*js"と書かれたグレーの正方形。マウスが押されると、テキストが消えます。');
    }

    // マウスが押されたときに全ての要素を削除します。

    }

    function draw() {
      // スライダーの値を使用して背景色を変更します。
      let g = slider.value();
      background(g);
    }

    // マウスがダブルクリックされたときに全ての要素を削除します。

      removeElements();
    }
