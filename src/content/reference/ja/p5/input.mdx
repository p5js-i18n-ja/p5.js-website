  <p>要素が入力を受け取ったときに関数を呼び出します。</p>

  <p><code>myElement.input()</code>は、テキスト入力やスライダーでよく使用されます。

  <code>myElement.input(false)</code>を呼び出すと、関数が無効になります。</p>

    <div>
    <code>
    let slider;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // スライダーを作成し、キャンバスの下に配置します。
      slider = createSlider(0, 255, 200);
      slider.position(0, 100);

      // スライダーが変更されたときにrepaint()を呼び出します。
      slider.input(repaint);

      describe('グレーの正方形と、その下にあるレンジスライダー。スライダーを動かすと背景のグレーの色合いが変化します。');
    }

    // スライダーの値を使用して背景を描画します。
    function repaint() {
      let g = slider.value();
      background(g);
    }

    let input;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 入力フィールドを作成し、キャンバスの下に配置します。
      input = createInput('');
      input.position(0, 100);

      // 入力が検出されたときにrepaint()を呼び出します。
      input.input(repaint);

      describe('グレーの正方形と、その下にあるテキスト入力バー。入力バーに書かれたテキストが正方形の中央に表示されます。');
    }

    // 背景をグレーに描画し、入力の値を表示します。
    function repaint() {
      background(200);
      let msg = input.value();
      text(msg, 5, 50);
    }
    </code>
