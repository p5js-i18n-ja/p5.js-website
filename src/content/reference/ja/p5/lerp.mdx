description: >
  <p>2つの数値の間の特定の増分で数値を計算します。</p>

  <p><code>amt</code>パラメータは、2つの数値の間を補間する量です。

  0.0は最初の数値に等しく、0.1は最初の数値に非常に近く、0.5は

  中間点で、1.0は2番目の数値に等しくなります。<code>lerp()</code>

  関数は、直線的な動きを作成したり、

  点線を描いたりするのに便利です。</p>

  <p><code>amt</code>の値が0未満または1より大きい場合、<code>lerp()</code>は

  元の区間外の数値を返します。例えば、

  <code>lerp(0, 10, 1.5)</code>を呼び出すと15が返されます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 座標の変数を宣言します。
      let a = 20;
      let b = 80;
      let c = lerp(a, b, 0.2);
      let d = lerp(a, b, 0.5);
      let e = lerp(a, b, 0.8);

      strokeWeight(5);

      // 元の点を黒で描画します。
      stroke(0);
      point(a, 50);
      point(b, 50);

      // 補間された点をグレーで描画します。
      stroke(100);
      point(c, 50);
      point(d, 50);
      point(e, 50);

      describe('水平線上に5つの点。外側の点は黒で、内側の点はグレーです。');
    }

      circle(x, y, 20);
    }

    // ユーザーがマウスをクリックしたときにxとyを設定します。
