description: >
  <p><code>String</code>型のシステム変数で、最後に入力されたキーの値を含みます。</p>

  <p>key変数は、

  <a href="https://en.wikipedia.org/wiki/ASCII#Printable_characters"
  target="_blank">ASCII</a>

  キーが入力されたかどうかを確認するのに役立ちます。例えば、<code>key === "a"</code>という式は、

  <code>a</code>キーが押された場合は<code>true</code>に、そうでない場合は<code>false</code>に

  評価されます。<code>key</code>は<code>LEFT_ARROW</code>や<code>ENTER</code>などの特殊キーでは

  更新されません。特殊キーにはkeyCodeを使用してください。複数の異なるキーの

  同時押しを確認するには、<a href="/reference/p5/keyIsDown/">keyIsDown()</a>関数を

  使用する必要があります。</p>

    // キャンバスをクリックしてキー入力の検出を開始します。

    let x = 50;
    let y = 50;

    function setup() {
      createCanvas(100, 100);

      background(200);

      describe(
        'グレーの正方形の中央に黒い円があります。ユーザーが "w"、"a"、"s"、"d" キーを押すと円が移動します。移動すると軌跡が残ります。'
      );
    }

    function draw() {
      // キーが押されたら x と y を更新します。
      if (keyIsPressed === true) {
        if (key === 'w') {
          y -= 1;
        } else if (key === 's') {
          y += 1;
        } else if (key === 'a') {
          x -= 1;
        } else if (key === 'd') {
          x += 1;
        }
      }

      // 円のスタイルを設定します。
      fill(0);

      // (x, y) に円を描画します。
      circle(x, y, 5);
    }
