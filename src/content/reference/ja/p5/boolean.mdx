description: >
  <p><code>String</code>または<code>Number</code>を<code>Boolean</code>に変換します。</p>

  <p><code>boolean()</code>は値を<code>true</code>または<code>false</code>に変換します。</p>

  <p>パラメータ<code>n</code>は変換する値です。<code>n</code>が文字列の場合、

  <code>boolean('true')</code>は<code>true</code>を返し、それ以外の文字列値は

  <code>false</code>を返します。<code>n</code>が数値の場合、<code>boolean(0)</code>は<code>false</code>を返し、

  それ以外の数値はすべて<code>true</code>を返します。配列が渡された場合、

  <code>boolean([0, 1, 'true', 'blue'])</code>のように、Boolean値の配列が

  返されます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 値の配列を作成します。
      let original = [0, 'hi', 123, 'true'];

      // 配列をBoolean値に変換します。
      let converted = boolean(original);

      // 変換されたBoolean値の配列を反復処理します。
      for (let i = 0; i < converted.length; i += 1) {

        // 変換された値に基づいて円のスタイルを設定します。
        if (converted[i] === true) {
          fill('blue');
        } else {
          fill('red');
        }

        // x座標を計算します。
        let x = (i + 1) * 20;

        // 円を描画します。
        circle(x, 50, 15);
      }

      describe(
        '灰色の背景に円の列。左の2つの円は赤で、右の2つの円は青。'
      );
    }
