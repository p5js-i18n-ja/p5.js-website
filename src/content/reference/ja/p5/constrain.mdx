      background(200);

      // 境界線を設定し、描画します。
      let leftWall = 25;
      let rightWall = 75;
      line(leftWall, 0, leftWall, 100);
      line(rightWall, 0, rightWall, 100);

      // マウスに自由に従う円を描画します。
      fill(255);
      circle(mouseX, 33, 9);

      // 制限された円を描画します。
      let xc = constrain(mouseX, leftWall, rightWall);
      fill(0);
      circle(xc, 67, 9);
    }
