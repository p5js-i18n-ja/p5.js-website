  <p><a href="/reference/p5/random/">random()</a>関数と

  <a href="/reference/p5/randomGaussian/">randomGaussian()</a>関数のシード値を設定します。</p>

  <p>デフォルトでは、<a href="/reference/p5/random/">random()</a>と

  <a href="/reference/p5/randomGaussian/">randomGaussian()</a>は、スケッチが実行されるたびに

  異なる結果を生成します。<code>randomSeed()</code>を定数の引数で呼び出すと、

  例えば<code>randomSeed(99)</code>のように、これらの関数はスケッチが

  実行されるたびに同じ結果を生成するようになります。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // ランダムな座標を取得
      let x = random(0, 100);
      let y = random(0, 100);

      // 白い円を描画
      circle(x, y, 10);

      // 一貫性のためにランダムシードを設定
      randomSeed(99);

      // ランダムな座標を取得
      x = random(0, 100);
      y = random(0, 100);

      // 黒い円を描画
      fill(0);
      circle(x, y, 10);

      describe('ランダムな位置に白い円が表示されます。黒い円は(27.4, 25.8)の位置に表示されます。');
    }
