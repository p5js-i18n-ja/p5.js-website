  <p>3Dスケッチの現在の（アクティブな）カメラを設定します。</p>

  <p><code>setCamera()</code>は、<a href="/reference/p5/createCamera/">createCamera()</a>で作成された
  複数のカメラ間で切り替えることができます。</p>

  <p>注意：<code>setCamera()</code>はWebGLモードでのみ使用できます。</p>

  - |-

    <div>
    <code>
    // ダブルクリックでカメラを切り替えます。

    let cam1;
    let cam2;
    let usingCam1 = true;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // 最初のカメラを作成します。
      // デフォルト設定のままにします。
      cam1 = createCamera();

      // 2番目のカメラを作成します。
      // 左上に配置します。
      // 原点を向くようにします。
      cam2 = createCamera();
      cam2.setPosition(400, -400, 800);
      cam2.lookAt(0, 0, 0);

      // 現在のカメラをcam1に設定します。
      setCamera(cam1);

      describe('グレーの背景に白い立方体。ユーザーがダブルクリックすると、カメラが正面図と俯瞰図の間で切り替わります。');
    }

    function draw() {
      background(200);

      // 箱を描画します。
      box();
    }

    // ユーザーがダブルクリックしたときに現在のカメラを切り替えます。
    function doubleClicked() {
      if (usingCam1 === true) {
        setCamera(cam2);
        usingCam1 = false;
      } else {
        setCamera(cam1);
        usingCam1 = true;
      }
    }
    </code>
