description: >
  <p>3Dスケッチ内の現在のカメラの位置と向きを設定します。</p>

  <p><code>camera()</code>は、オブジェクトを異なる角度から見ることを可能にします。
  9つのパラメータがあり、すべてオプションです。</p>

  <p>最初の3つのパラメータ、<code>x</code>、<code>y</code>、<code>z</code>は、
  カメラの位置の座標です。例えば、<code>camera(0, 0, 0)</code>を呼び出すと、
  カメラを原点<code>(0, 0, 0)</code>に配置します。デフォルトでは、カメラは
  <code>(0, 0, 800)</code>に配置されています。</p>

  <p>次の3つのパラメータ、<code>centerX</code>、<code>centerY</code>、<code>centerZ</code>は、
  カメラが向いている点の座標です。例えば、<code>camera(0, 0, 0, 10, 20, 30)</code>を
  呼び出すと、カメラを原点<code>(0, 0, 0)</code>に配置し、<code>(10, 20, 30)</code>に
  向けます。デフォルトでは、カメラは原点<code>(0, 0, 0)</code>を向いています。</p>

  <p>最後の3つのパラメータ、<code>upX</code>、<code>upY</code>、<code>upZ</code>は、
  「上」ベクトルの成分です。「上」ベクトルはカメラのy軸の向きを決定します。例えば、
  <code>camera(0, 0, 0, 10, 20, 30, 0, -1, 0)</code>を呼び出すと、カメラを
  原点<code>(0, 0, 0)</code>に配置し、<code>(10, 20, 30)</code>に向け、「上」ベクトルを
  <code>(0, -1, 0)</code>に設定します。これは上下逆さまに持っているようなものです。
  デフォルトでは、「上」ベクトルは<code>(0, 1, 0)</code>です。</p>

  <p>注意：<code>camera()</code>はWebGLモードでのみ使用できます。</p>

    let xSlider;
    let ySlider;
    let zSlider;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // カメラの座標を設定するスライダーオブジェクトを作成する
      xSlider = createSlider(-400, 400, 400);
      xSlider.position(0, 100);
      xSlider.size(100);
      ySlider = createSlider(-400, 400, -200);
      ySlider.position(0, 120);
      ySlider.size(100);
      zSlider = createSlider(0, 1600, 800);
      zSlider.position(0, 140);
      zSlider.size(100);

      describe(
        '灰色の背景に描かれた白い立方体。画像の下に3つのスライダーが表示される。ユーザーがスライダーを動かすとカメラの位置が変わる。'
      );
    }

    function draw() {
      background(200);

      // スライダーからカメラの座標を取得する
      let x = xSlider.value();
      let y = ySlider.value();
      let z = zSlider.value();

      // カメラを移動する
      camera(x, y, z);

      // 箱を描画する
      box();
    }
