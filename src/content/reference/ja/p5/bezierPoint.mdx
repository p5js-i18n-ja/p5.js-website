description: >
  <p>補間を使用してベジエ曲線に沿った座標を計算します。</p>

  <p><code>bezierPoint()</code>は、アンカーポイントと制御点を使用して

  ベジエ曲線に沿った座標を計算します。<a href="/reference/p5/bezier/">bezier()</a>関数と

  同じ順序でポイントを指定することが期待されます。<code>bezierPoint()</code>は

  一度に1つの軸に対して機能します。アンカーポイントと制御点の

  x座標を渡すと、曲線上のポイントのx座標が計算されます。

  y座標を渡すと、曲線上のポイントのy座標が計算されます。</p>

  <p>最初のパラメータ<code>a</code>は、最初のアンカーポイントの座標です。</p>

  <p>2番目と3番目のパラメータ<code>b</code>と<code>c</code>は、制御点の座標です。</p>

  <p>4番目のパラメータ<code>d</code>は、最後のアンカーポイントの座標です。</p>

  <p>5番目のパラメータ<code>t</code>は、曲線に沿って補間する量です。0は

  最初のアンカーポイント、1は2番目のアンカーポイント、0.5はその中間点です。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 曲線のアンカーポイントと制御点の座標を設定します。
      let x1 = 85;
      let x2 = 10;
      let x3 = 90;
      let x4 = 15;
      let y1 = 20;
      let y2 = 10;
      let y3 = 90;
      let y4 = 80;

      // 曲線のスタイルを設定します。
      noFill();

      // 曲線を描画します。
      bezier(x1, y1, x2, y2, x3, y3, x4, y4);

      // 曲線のパスに沿って円を描画します。
      fill(255);

      // 右上。
      let x = bezierPoint(x1, x2, x3, x4, 0);
      let y = bezierPoint(y1, y2, y3, y4, 0);
      circle(x, y, 5);

      // 中央。
      x = bezierPoint(x1, x2, x3, x4, 0.5);
      y = bezierPoint(y1, y2, y3, y4, 0.5);
      circle(x, y, 5);

      // 左下。
      x = bezierPoint(x1, x2, x3, x4, 1);
      y = bezierPoint(y1, y2, y3, y4, 1);
      circle(x, y, 5);

      describe('グレーの正方形上に黒のS字カーブ。カーブの端点と中心が白い円で示されています。');
    }

    function setup() {
      createCanvas(100, 100);

      describe('グレーの正方形上に黒のS字カーブ。白い円がカーブに沿って前後に動きます。');
    }

    function draw() {
      background(200);

      // 曲線のアンカーポイントと制御点の座標を設定します。
      let x1 = 85;
      let x2 = 10;
      let x3 = 90;
      let x4 = 15;
      let y1 = 20;
      let y2 = 10;
      let y3 = 90;
      let y4 = 80;

      // 曲線を描画します。
      noFill();
      bezier(x1, y1, x2, y2, x3, y3, x4, y4);

      // 円の座標を計算します。
      let t = 0.5 * sin(frameCount * 0.01) + 0.5;
      let x = bezierPoint(x1, x2, x3, x4, t);
      let y = bezierPoint(y1, y2, y3, y4, t);

      // 円を描画します。
      fill(255);
      circle(x, y, 5);
    }
