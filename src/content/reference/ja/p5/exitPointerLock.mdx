description: >
  <p><a href="/reference/p5/requestPointerLock/">requestPointerLock</a>で開始されたポインターロックを終了します。</p>

  <p><code>requestPointerLock()</code>を呼び出すと、
  <a href="/reference/p5/mouseX/">mouseX</a>、<a href="/reference/p5/mouseY/">mouseY</a>、
  <a href="/reference/p5/pmouseX/">pmouseX</a>、および<a href="/reference/p5/pmouseY/">pmouseY</a>の値がロックされます。
  <code>exitPointerLock()</code>を呼び出すと、マウスシステム変数の更新が再開されます。</p>

  <p>注意：ほとんどのブラウザでは、<code>requestPointerLock()</code>を呼び出す前に
  クリックなどの入力が必要です。<code>requestPointerLock()</code>は
  <a href="/reference/p5/doubleClicked/">doubleClicked()</a>などのイベント関数内で
  呼び出すことをお勧めします。</p>

    <code>
    let isLocked = false;

    function setup() {
      createCanvas(100, 100);

      describe(
        '中央に単語が表示されている灰色の正方形。ユーザーがダブルクリックすると、単語が"Unlocked"と"Locked"の間で切り替わります。'
      );
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER);
      textSize(16);

      // ポインターがロックされているかどうかをユーザーに伝えます。
      if (isLocked === true) {
        text('Locked', 50, 50);
      } else {
        text('Unlocked', 50, 50);
      }
    }

    // ユーザーがダブルクリックしたときにポインターロックを切り替えます。
    function doubleClicked() {
      if (isLocked === true) {
        exitPointerLock();
        isLocked = false;
      } else {
        requestPointerLock();
        isLocked = true;
      }
    }
