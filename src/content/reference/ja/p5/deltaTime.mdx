    let x = 0;
    let speed = 0.05;

    function setup()  {
      createCanvas(100, 100);

      // フレームレートを30に設定します。
      frameRate(30);

      describe('白い円が灰色の背景上を左から右に移動します。右端に達すると左側に再び現れます。');
    }

    function draw() {
      background(200);

      // deltaTimeを使用して
      // 位置の変化を計算します。
      let deltaX = speed * deltaTime;

      // x変数を更新します。
      x += deltaX;

      // xが100より大きい場合、
      // 0にリセットします。
      if (x > 100)  {
        x = 0;
      }

      // xを使用して円の
      // 位置を設定します。
      circle(x, 50, 20);
