description: |
  <p>現在のフォントの現在のサイズでのディセントを計算します。</p>
  <p>ディセントは、ベースラインより下の最も低い文字の距離をピクセル単位で表します。</p>
line: 395
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let font;

    function preload()  {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup()  {
      createCanvas(100, 100);

      background(200);

      // テキストのスタイルを設定します。
      textFont(font);

      // フォントごとに異なります。
      let fontScale = 0.8;

      let baseY = 25;
      strokeWeight(0.5);

      // 小さいテキストを描画します。
      textSize(24);
      text('dp', 0, baseY);

      // ベースラインとディセントを描画します。
      let d = textDescent() * fontScale;
      line(0, baseY, 23, baseY);
      line(23, baseY, 23, baseY + d);

      // 大きいテキストを描画します。
      textSize(48);
      text('dp', 45, baseY);

      // ベースラインとディセントを描画します。
      d = textDescent() * fontScale;
      line(45, baseY, 91, baseY);
      line(91, baseY, 91, baseY + d);

      describe('異なるサイズで2回書かれた文字"dp"。各バージョンには水平のベースラインがあります。各ベースラインから"p"の下部まで垂直線が伸びています。');
    }
    </code>
    </div>
class: p5

  description: ピクセル単位で測定されたディセント。
