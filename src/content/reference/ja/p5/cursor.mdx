description: >
  <p>カーソルの外観を変更します。</p>

  <p>最初のパラメータ<code>type</code>は、表示するカーソルの種類を設定します。

  組み込みのオプションには<code>ARROW</code>、<code>CROSS</code>、
  <code>HAND</code>、<code>MOVE</code>、<code>TEXT</code>、
  <code>WAIT</code>があります。

  <code>cursor()</code>は、文字列として渡される標準的なCSSカーソルプロパティも

  認識します：<code>'help'</code>、<code>'wait'</code>、<code>'crosshair'</code>、
  <code>'not-allowed'</code>、<code>'zoom-in'</code>、

  <code>'grab'</code>などです。画像へのパスが渡された場合、例えば

  <code>cursor('/assets/target.png')</code>のように、その画像がカーソルとして使用されます。

  画像は.cur、.gif、.jpg、.jpeg、または.png形式である必要があり、<a
  href="https://developer.mozilla.org/en-US/docs/Web/CSS/cursor#icon_size_limits">最大で32×32ピクセルの大きさ</a>でなければなりません。</p>

  <p>パラメータ<code>x</code>と<code>y</code>はオプションです。画像がカーソルとして

  使用される場合、<code>x</code>と<code>y</code>は画像内で指し示す位置を設定します。

  デフォルトではどちらも0で、カーソルは画像の左上隅を指します。
  <code>x</code>

  と<code>y</code>は、それぞれ画像の幅と高さより小さくなければなりません。</p>

    <code>
    function setup() {
      createCanvas(100, 100);

      describe('4つの区画に分かれた灰色の正方形。マウスが各区画に移動すると、カーソルの画像が変化します。');
    }

    function draw() {
      background(200);

      // キャンバスを4つの区画に分割。
      line(50, 0, 50, 100);
      line(0, 50, 100, 50);

      // マウスの位置に基づいてカーソルを変更。
      if (mouseX < 50 && mouseY < 50) {
        cursor(CROSS);
      } else if (mouseX > 50 && mouseY < 50) {
        cursor('progress');
      } else if (mouseX > 50 && mouseY > 50) {
        cursor('https://avatars0.githubusercontent.com/u/1617169?s=16');
      } else {
        cursor('grab');
      }
    }
    </code>
    </div>

      createCanvas(100, 100);

      describe('3つの紫色の曲線の画像がマウスに従います。マウスが押されると画像がシフトします。');
    }

    function draw() {
      background(200);

      // マウスが押されたときに

      } else {
        cursor('https://avatars0.githubusercontent.com/u/1617169?s=16');

    }
