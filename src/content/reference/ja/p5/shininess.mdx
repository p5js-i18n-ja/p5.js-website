description: >
  <p><a href="/reference/p5/specularMaterial/">specularMaterial()</a>の光沢（「艶」）の量を設定します。</p>

  <p>光沢のある材質は、光沢のない材質よりも反射光を集中させます。
  <code>shininess()</code>は、
  <a href="/reference/p5/directionalLight/">directionalLight()</a>、
  <a href="/reference/p5/pointLight/">pointLight()</a>、
  <a href="/reference/p5/spotLight/">spotLight()</a>を含む光源の反射方法に影響します。</p>

  <p>パラメータ<code>shine</code>は、光沢の量を設定する数値です。
  <code>shine</code>は1より大きくなければなりません。デフォルト値は1です。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);

      describe(
        '灰色の背景に2つの赤い球体が描かれています。マウスが動くと、表面に白い光が反射します。右の球体は左の球体よりも光沢があります。'
      );
    }

    function draw() {
      background(200);

      // 赤いアンビエントライトをオンにする
      ambientLight(255, 0, 0);

      // マウスの座標を取得
      let mx = mouseX - 50;
      let my = mouseY - 50;

      // マウスに追従する白いポイントライトをオンにする
      pointLight(255, 255, 255, mx, my, 50);

      // 球体のスタイルを設定
      noStroke();

      // グレースケール値でスペキュラーマテリアルを追加
      specularMaterial(255);

      // 低い光沢の左の球体を描画
      translate(-25, 0, 0);
      shininess(10);
      sphere(20);

      // 高い光沢の右の球体を描画
      translate(50, 0, 0);
      shininess(100);
      sphere(20);
    }
    </code>
