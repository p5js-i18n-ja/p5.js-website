    let img;

    // 画像を読み込む
    function preload() {
      img = loadImage('/assets/rockies.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      // 画像を表示する
      image(img, 0, 0, 100, 100);

      // ピクセル密度を取得する
      let d = pixelDensity();

      // pixels配列の半分のインデックスを計算する
      let halfImage = 4 * (d * width) * (d * height / 2);

      // pixels配列を読み込む
      loadPixels();

      // キャンバスの上半分を下半分にコピーする
      for (let i = 0; i < halfImage; i += 1) {
        pixels[i + halfImage] = pixels[i];
      }

      // キャンバスを更新する
      updatePixels();

      describe('2つの同一の山岳風景画像が、上下に重なっている。');
    }
