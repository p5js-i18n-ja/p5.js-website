description: >
  <p>ドロップダウンメニュー <code><select></select></code> 要素を作成します。</p>

  <p>パラメータはオプションです。<code>true</code> が渡された場合、

  <code>let mySelect = createSelect(true)</code> のように、ドロップダウンは

  複数選択をサポートします。既存の <code><select></select></code> 要素が

  渡された場合、<code>let mySelect = createSelect(otherSelect)</code> のように、既存の

  要素は新しい <a href="/reference/p5/p5.Element">p5.Element</a>

  オブジェクトでラップされます。</p>

  <p>ドロップダウンは <a href="/reference/p5/p5.Element">p5.Element</a> クラスを拡張し、

  オプションを管理するためのいくつかの便利なメソッドを提供します：</p>

  <ul>

  <li><code>mySelect.option(name, [value])</code> はメニューにオプションを追加します。
  最初のパラメータ <code>name</code> は、オプションの名前と値を設定する文字列です。2番目のパラメータ <code>value</code> はオプションです。提供された場合、キー <code>name</code> に対応する値を設定します。<code>name</code> を持つオプションが既に存在する場合、その値は <code>value</code> に変更されます。</li>

  <li><code>mySelect.value()</code> は現在選択されているオプションの値を返します。</li>

  <li><code>mySelect.selected()</code> は現在選択されているオプションを返します。</li>

  <li><code>mySelect.selected(option)</code> は指定されたオプションをデフォルトで選択します。</li>

  <li><code>mySelect.disable()</code> はドロップダウン要素全体を無効としてマークします。</li>

  <li><code>mySelect.disable(option)</code> は指定されたオプションを無効としてマークします。</li>

  <li><code>mySelect.enable()</code> はドロップダウン要素全体を有効としてマークします。</li>

  <li><code>mySelect.enable(option)</code> は指定されたオプションを有効としてマークします。</li>

  </ul>

    // WindowsとLinuxではCTRLキーを押したまま、
    // macOSではCMDキーを押したまま複数のオプションを選択できます。
    let mySelect;

    function setup() {
      createCanvas(100, 100);

      // ドロップダウンを作成し、複数選択を許可します。
      // キャンバスの下に配置します。
      mySelect = createSelect(true);
      mySelect.position(0, 100);

      // 色のオプションを追加します。
      mySelect.option('red');
      mySelect.option('green');
      mySelect.option('blue');
      mySelect.option('yellow');

      describe('灰色の正方形とその下にドロップダウンメニューがあります。色が選択されると、その色の円が表示されます。');
    }

    function draw() {
      background(200);

      // 選択された値を使用して円を描画します。
      let colors = mySelect.selected();
      for (let i = 0; i < colors.length; i += 1) {
        // x座標を計算します。
        let x = 10 + i * 20;

        // 色にアクセスします。
        let c = colors[i];

        // 円を描画します。
        fill(c);
        circle(x, 50, 20);
      }
    }
