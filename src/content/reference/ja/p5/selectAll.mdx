  <p>与えられた<a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">CSSセレクタ文字列</a>に一致するすべての要素をページ内で検索します。</p>

  <p>セレクタ文字列はID、クラス、タグ名、またはそれらの組み合わせが可能です。

  <code>selectAll()</code>は、一致するものが見つかった場合は<a
  href="/reference/p5/p5.Element">p5.Element</a>オブジェクトの配列を返し、

  見つからなかった場合は空の配列を返します。</p>

  <p>2番目のパラメータ<code>container</code>はオプションです。これは検索対象となる

  コンテナを指定します。<code>container</code>はCSSセレクタ文字列、

  <a href="/reference/p5/p5.Element">p5.Element</a>オブジェクト、または

  <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement/"
  target="_blank">HTMLElement</a>オブジェクトのいずれかです。</p>

    function setup() {
      // 3つのボタンを作成し、配置します。
      let b1 = createButton('1');
      b1.position(0, 0);
      let b2 = createButton('2');
      b2.position(0, 30);
      let b3 = createButton('3');
      b3.position(0, 60);

      // 各ボタンにクラス属性を追加します。
      b1.class('btn');
      b2.class('btn btn-pink');
      b3.class('btn');

      // クラスでボタンを選択します。
      let buttons = selectAll('.btn');
      let pinkButtons = selectAll('.btn-pink');

      // 選択したボタンにスタイルを適用します。
      buttons.forEach(setFont);
      pinkButtons.forEach(setColor);

      describe('垂直に積み重ねられた3つのボタン。ボタンには「1」、「2」、「3」とラベルが付いています。ボタン「1」と「3」は灰色です。ボタン「2」はピンク色です。');
    }

    // ボタンのフォントをComic Sans MSに設定します。
    function setFont(btn) {
      btn.style('font-family', 'Comic Sans MS');
    }

    // ボタンの背景色とフォント色を設定します。
    function setColor(btn) {
      btn.style('background', 'deeppink');
      btn.style('color', 'white');
    }
    </code>
