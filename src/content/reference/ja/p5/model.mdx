  <p><a href="/reference/p5/p5.Geometry">p5.Geometry</a>オブジェクトをキャンバスに描画します。</p>

  <p>パラメータ<code>model</code>は、描画する<a href="/reference/p5/p5.Geometry">p5.Geometry</a>オブジェクトです。

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a>オブジェクトは、<a href="/reference/p5/buildGeometry/">buildGeometry()</a>や

  <a href="/reference/p5/beginGeometry/">beginGeometry()</a>と<a href="/reference/p5/endGeometry/">endGeometry()</a>で作成できます。また、

  <a href="/reference/p5/loadGeometry/">loadGeometry()</a>でファイルから読み込むこともできます。</p>

  <p>注意: <code>model()</code>はWebGLモードでのみ使用できます。</p>

    let shape;

    // マウスをクリックしてドラッグすると、シーンを異なる角度から見ることができます。

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // p5.Geometryオブジェクトを作成します。
      shape = buildGeometry(createArrow);

      describe('灰色の背景に描かれた2つの白い矢印。右側の矢印はゆっくりと回転します。');
    }

    function draw() {
      background(50);

      // マウスでのオービットを有効にします。
      orbitControl();

      // ライトをオンにします。
      lights();

      // 矢印のスタイルを設定します。
      noStroke();

      // p5.Geometryオブジェクトを描画します。
      model(shape);

      // 座標系を移動および回転させます。
      translate(30, 0, 0);
      rotateZ(frameCount * 0.01);

      // p5.Geometryオブジェクトを再度描画します。
      model(shape);
    }

    function createArrow() {
      // p5.Geometryオブジェクトに形状を追加します。
      push();
      rotateX(PI);
      cone(10);
      translate(0, -10, 0);
      cylinder(3, 20);
      pop();
    }
    </code>
