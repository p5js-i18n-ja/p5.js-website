description: >
  <p><a href="/reference/p5/specularMaterial/">specularMaterial()</a>の「金属性」の量を設定します。</p>

  <p><code>metalness()</code>は、材料をより金属的に見せることができます。これは、
  材料が光源を反射する方法に影響を与えます。以下の光源の反射に影響します：

  <a href="/reference/p5/directionalLight/">directionalLight()</a>、

  <a href="/reference/p5/pointLight/">pointLight()</a>、

  <a href="/reference/p5/spotLight/">spotLight()</a>、そして

  <a href="/reference/p5/imageLight/">imageLight()</a>。</p>

  <p>パラメータ<code>metallic</code>は、金属性の量を設定する数値です。
  <code>metallic</code>は1より大きくなければなりません（1がデフォルト値です）。
  <code>metalness(100)</code>のようなより高い値は、鏡面反射材料をより
  金属的に見せます。</p>

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe(
        '灰色の背景に2つの青い球体が描かれています。マウスが動くと、その表面に白い光が反射します。右の球体は左の球体よりも金属性が高くなっています。'
      );
    }

    function draw() {
      background(200);

      // 環境光をオンにします。
      ambientLight(200);

      // マウスの座標を取得します。
      let mx = mouseX - 50;
      let my = mouseY - 50;

      // マウスに追従する白い点光源をオンにします。
      pointLight(255, 255, 255, mx, my, 50);

      // 球体のスタイルを設定します。
      noStroke();
      fill(30, 30, 255);
      specularMaterial(255);
      shininess(20);

      // 低い金属性で左の球体を描画します。
      translate(-25, 0, 0);
      metalness(1);
      sphere(20);

      // 高い金属性で右の球体を描画します。
      translate(50, 0, 0);
      metalness(50);
      sphere(20);
    }
    </code>
