  <p><a href="/reference/p5/noise/">noise()</a>関数のシード値を設定します。</p>

  <p>デフォルトでは、<a href="/reference/p5/noise/">noise()</a>は
  スケッチを実行するたびに異なる結果を生成します。
  <code>noiseSeed()</code>を定数引数で呼び出すと、
  例えば<code>noiseSeed(99)</code>のように、<a href="/reference/p5/noise/">noise()</a>は
  スケッチを実行するたびに同じ結果を生成するようになります。</p>

  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // 一貫した結果を得るためにノイズシードを設定します。
      noiseSeed(99);

      describe('ランダムに成長する黒い長方形。最初は右に、次に左に成長します。');
    }

    function draw() {
      // ノイズレベルとスケールを設定します。
      let noiseLevel = 100;
      let noiseScale = 0.005;

      // 入力座標をスケーリングします。
      let nt = noiseScale * frameCount;

      // ノイズ値を計算します。
      let x = noiseLevel * noise(nt);

      // 線を描画します。
      line(x, 0, x, height);
    }
