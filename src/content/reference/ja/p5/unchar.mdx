  <p>単一文字の<code>String</code>を<code>Number</code>に変換します。</p>

  <p><code>unchar()</code>は単一文字の文字列を対応する整数（整数）に変換します。</p>

  <p>パラメータ<code>n</code>は変換する文字です。例えば、<code>unchar('A')</code>は数字65を返します。

  配列が渡された場合、例えば<code>unchar(['A', 'B', 'C'])</code>のように、整数の配列が返されます。</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // 文字の配列を作成します。
      let original = ['A', 'B', 'C'];

      // 文字列を数値に変換します。
      let converted = unchar(original);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textSize(16);

      // 変換された配列の要素を反復処理します。
      for (let i = 0; i < converted.length; i += 1) {

        // y座標を計算します。
        let y = (i + 1) * 25;

        // 元の値と変換後の値を表示します。
        text(`${original[i]} : ${converted[i]}`, 50, y);
      }

      describe(
        '3行に分けて書かれた "A : 65"、"B : 66"、"C :67" というテキスト。テキストは灰色の背景に黒で書かれています。'
      );
    }
