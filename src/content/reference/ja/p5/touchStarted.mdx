description: >
  <p>ユーザーが画面にタッチするたびに1回呼び出される関数です。</p>

  <p><code>touchStarted()</code>という関数を宣言すると、ユーザーがタッチスクリーンデバイスでタッチを開始するたびに
  自動的に実行されるコードブロックを設定します：</p>

  <pre><code class="language-js">function touchStarted() {
    // 実行するコード
  }

  </code></pre>

  <p><code>touchStarted()</code>がp5.jsによって呼び出されると、<a href="/reference/p5/touches/">touches</a>配列が
  最新のタッチポイントで更新されます：</p>

  <pre><code class="language-js">function touchStarted() {
    // 背景を塗りつぶす
    background(200);

    // 各タッチポイントを1回だけ円で表示
    for (let touch of touches) {
      circle(touch.x, touch.y, 40);
    }
  }

  </code></pre>

  <p>パラメータのeventはオプションです。<code>touchStarted()</code>には
  タッチイベントを説明するプロパティを持つ
  <a href="https://developer.mozilla.org/en-US/docs/Web/API/TouchEvent/"
  target="_blank">TouchEvent</a>
  オブジェクトが渡されます：</p>

  <pre><code class="language-js">function touchStarted(event) {
    // イベントを使用するコード
    console.log(event);
  }

  </code></pre>

  <p>タッチスクリーンデバイスでは、<code>touchStarted()</code>が宣言されていない場合、
  ユーザーのタッチが開始されると<a href="/reference/p5/mousePressed/">mousePressed()</a>が
  実行されます。<code>touchStarted()</code>が宣言されている場合、ユーザーのタッチが
  開始されると<code>touchStarted()</code>が実行され、<a href="/reference/p5/mousePressed/">mousePressed()</a>
  は実行されません。</p>

  <p>注意：<code>touchStarted()</code>、<a href="/reference/p5/touchEnded/">touchEnded()</a>、
  <a href="/reference/p5/touchMoved/">touchMoved()</a>はすべて関連しています。
  <code>touchStarted()</code>はユーザーがタッチスクリーンデバイスにタッチした瞬間に実行されます。
  <a href="/reference/p5/touchEnded/">touchEnded()</a>はユーザーがタッチを終了した瞬間に実行されます。
  <a href="/reference/p5/touchMoved/">touchMoved()</a>はユーザーがタッチポイントを
  移動させている間、繰り返し実行されます。</p>

    // 同時にキャンバスにタッチしてください。

    let bgColor = 50;
    let fillColor = 255;
    let borderWidth = 0.5;

    function setup() {
      createCanvas(100, 100);

      describe(
        '上部中央に数字0が表示された灰色の正方形。数字はユーザーが画面にタッチしている箇所の数を追跡します。各タッチポイントに円が表示され、イベントに応じてスタイルが変化します。'
      );
    }

    function draw() {
      background(bgColor);

      // テキストのスタイルを設定
      textAlign(CENTER);
      textSize(16);
      fill(0);
      noStroke();

      // タッチポイントの数を表示
      text(touches.length, 50, 20);

      // タッチポイントのスタイルを設定
      fill(fillColor);
      stroke(0);
      strokeWeight(borderWidth);

      // タッチポイントを円として表示
      for (let touch of touches) {
        circle(touch.x, touch.y, 40);
      }
    }

    // 背景色をランダムなグレースケール値に設定
    function touchStarted() {
      bgColor = random(80, 255);
    }

    // 塗りつぶし色をランダムなグレースケール値に設定
    function touchEnded() {
      fillColor = random(0, 255);
    }

    // ストロークの太さを設定
    function touchMoved() {
      // 境界線の幅を増加
      borderWidth += 0.1;

      // 境界線が太すぎる場合、リセット
      if (borderWidth > 20) {
        borderWidth = 0.5;
      }
    }
