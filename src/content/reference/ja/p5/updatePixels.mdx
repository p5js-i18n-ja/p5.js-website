  <p><a href="/reference/p5/pixels/">pixels</a>配列内のRGBA値でキャンバスを更新します。</p>

  <p><code>updatePixels()</code>は、<a href="/reference/p5/pixels/">pixels</a>配列の値を変更した後にのみ呼び出す必要があります。このような変更は、<a href="/reference/p5/loadPixels/">loadPixels()</a>を呼び出した直後、または<a href="/reference/p5/set/">set()</a>を呼び出すことで行うことができます。</p>

example:
  - |-

    <div>
    <code>
    let img;

    // 画像を読み込む
    function preload() {
      img = loadImage('/assets/rockies.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      // 画像を表示
      image(img, 0, 0, 100, 100);

      // ピクセル密度を取得
      let d = pixelDensity();

      // pixels配列の半分のインデックスを計算
      let halfImage = 4 * (d * width) * (d * height / 2);

      // pixels配列を読み込む
      loadPixels();

      // キャンバスの上半分を下半分にコピー
      for (let i = 0; i < halfImage; i += 1) {
        pixels[i + halfImage] = pixels[i];
      }

      // キャンバスを更新
      updatePixels();

      describe('2つの同一の山岳風景画像が、上下に重なっている。');
    }
    </code>
