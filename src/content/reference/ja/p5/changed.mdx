    let dropdown;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // ドロップダウンメニューを作成し、いくつかの色オプションを追加します。
      dropdown = createSelect();
      dropdown.position(0, 0);
      dropdown.option('red');
      dropdown.option('green');
      dropdown.option('blue');

      // 色オプションが変更されたときにpaintBackground()を呼び出します。
      dropdown.changed(paintBackground);

      describe('上部にドロップダウンメニューがある灰色の正方形。オプションが選択されると正方形の色が変わります。');
    }

    // 選択された色で背景を塗ります。
    function paintBackground() {
      let c = dropdown.value();
      background(c);
    }

    let checkbox;

    function setup() {
      createCanvas(100, 100);

      background(200);

      // チェックボックスを作成し、キャンバスの下に配置します。
      checkbox = createCheckbox(' circle');
      checkbox.position(0, 100);

      // チェックボックスが変更されたときにrepaint()を呼び出します。
      checkbox.changed(repaint);

      describe('下に「circle」と書かれたチェックボックスがある灰色の正方形。ボックスがチェックされると白い円が表示され、そうでない場合は消えます。');
    }

    // 背景を灰色に塗り、円を描くかどうかを決定します。
    function repaint() {
      background(200);
      if (checkbox.checked() === true) {
        circle(50, 50, 30);
      }
    }
    </code>
