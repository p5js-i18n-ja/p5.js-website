description: >
  <p>カスタム形状に頂点を追加します。</p>

  <p><code>vertex()</code>は、<a href="/reference/p5/beginShape/">beginShape()</a>と

  <a href="/reference/p5/endShape/">endShape()</a>関数の間で描画される頂点の

  座標を設定します。</p>

  <p>最初の2つのパラメータ、<code>x</code>と<code>y</code>は、頂点のx座標とy座標を

  設定します。</p>

  <p>3番目のパラメータ、<code>z</code>はオプションです。WebGLモードで頂点の

  z座標を設定します。デフォルトでは、<code>z</code>は0です。</p>

  <p>4番目と5番目のパラメータ、<code>u</code>と<code>v</code>もオプションです。これらは

  <a href="/reference/p5/endShape/">endShape()</a>と共に使用される場合の頂点の

  テクスチャのu座標とv座標を設定します。デフォルトでは、<code>u</code>と<code>v</code>は

  ともに0です。</p>

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('灰色の背景に白い四角錐が描かれています。四角錐の表面には赤、緑、青、黄色のグラデーションがあります。');
    }

    function draw() {
      background(200);

      // マウスでの軌道制御を有効にします。
      orbitControl();

      // テクスチャを作成します。
      let img = createImage(2, 2);
      img.loadPixels();
      img.set(0, 0, color(255, 0, 0));
      img.set(1, 0, color(0, 255, 0));
      img.set(0, 1, color(0, 0, 255));
      img.set(1, 1, color(255, 255, 0));
      img.updatePixels();

      // テクスチャを適用します。
      texture(img);

      // 形状の描画を開始します。
      beginShape();

      // 頂点を追加します。
      vertex(-40, -40, 0, 0, 0);
      vertex(40, -40, 0, 1, 0);
      vertex(40, 40, 0, 1, 1);
      vertex(-40, 40, 0, 0, 1);
      vertex(0, 0, 50, 0.5, 0.5);

      // 形状の描画を停止します。
      endShape(CLOSE);
    }
    </code>
