  <p>現在のフォントの現在のサイズでのアセントを計算します。</p>
  <p>アセントは、ベースラインより上の最も高い文字の距離をピクセル単位で表します。</p>
line: 337
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let font;

    function preload()  {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup()  {
      createCanvas(100, 100);

      background(200);

      // テキストのスタイルを設定します。
      textFont(font);

      // フォントごとに異なります。
      let fontScale = 0.8;

      let baseY = 75;
      strokeWeight(0.5);

      // 小さいテキストを描画します。
      textSize(24);
      text('dp', 0, baseY);

      // ベースラインとアセントを描画します。
      let a = textAscent() * fontScale;
      line(0, baseY, 23, baseY);
      line(23, baseY - a, 23, baseY);

      // 大きいテキストを描画します。
      textSize(48);
      text('dp', 45, baseY);

      // ベースラインとアセントを描画します。
      a = textAscent() * fontScale;
      line(45, baseY, 91, baseY);
      line(91, baseY - a, 91, baseY);

      describe('異なるサイズで2回書かれた文字"dp"。各バージョンには水平のベースラインがあります。各ベースラインから"d"の上部まで垂直線が伸びています。');
    }
    </code>
    </div>
