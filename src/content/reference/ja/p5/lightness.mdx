description: >
  <p>色の明度の値を取得します。</p>

  <p><code>lightness()</code>は、<a href="/reference/p5/p5.Color">p5.Color</a>オブジェクト、
  色成分の配列、またはCSSの色文字列からHSLの明度の値を抽出します。</p>

  <p>デフォルトでは、<code>lightness()</code>は色のHSL明度を0から100の範囲で返します。
  <a href="/reference/colorMode/">colorMode()</a>がHSLに設定されている場合、
  指定された範囲内で明度の値を返します。</p>

  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(50);

      // HSL色を使用
      colorMode(HSL);

      // HSL値を使用してp5.Colorオブジェクトを作成
      let c = color(0, 100, 75);

      // 左の長方形を描画
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // 'lightValue'を75に設定
      let lightValue = lightness(c);

      // 右の長方形を描画
      fill(lightValue);
      rect(50, 15, 35, 70);

      describe('2つの長方形。左側はサーモンピンク、右側はグレー。');
    }
