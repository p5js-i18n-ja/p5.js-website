description: >
  <p>ウェブブラウザのローカルストレージから値を取得します。</p>

  <p>ウェブブラウザは、組み込みの
  <a
  href="https://developer.mozilla.org/en-US/docs/Web/API/Window/localStorage/"
  target="_blank">localStorage オブジェクト</a>
  を使用して少量のデータを保存できます。
  <code>localStorage</code>に保存されたデータは、
  ページの更新やブラウザの再起動後でも、いつでも取得できます。データはキーと値のペアとして保存されます。</p>

  <p><a href="/reference/p5/storeItem/">storeItem()</a>を使用すると
  <code>localStorage</code>に値を簡単に保存でき、<code>getItem()</code>を使用すると
  簡単に取得できます。</p>

  <p>最初のパラメータ<code>key</code>は、保存される値の名前を
  文字列として指定します。</p>

  <p>2番目のパラメータ<code>value</code>は、取得する値を
  文字列として指定します。例えば、<code>getItem('size')</code>を呼び出すと、
  キー<code>size</code>に関連付けられた値を取得します。</p>

  <p>注意：パスワードや個人情報などの機密データは
  <code>localStorage</code>に保存しないでください。</p>

example:
  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      // プレイヤーの名前を保存します。
      storeItem('name', 'Feist');

      // プレイヤーのスコアを保存します。
      storeItem('score', 1234);

      describe('灰色の背景に黒で "Feist: 1234" と書かれたテキスト。');
    }

    function draw() {
      background(200);

      // テキストのスタイルを設定します。
      textAlign(CENTER, CENTER);
      textSize(14);

      // 名前を取得します。
      let name = getItem('name');

      // スコアを取得します。
      let score = getItem('score');

      // スコアを表示します。
      text(`${name}: ${score}`, 50, 50);
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      // オブジェクトを作成します。
      let p = { x: 50, y: 50 };

      // オブジェクトを保存します。
      storeItem('position', p);

      describe('灰色の背景に白い円。');
    }

    function draw() {
      background(200);

      // オブジェクトを取得します。
      let p = getItem('position');

      // 円を描画します。
      circle(p.x, p.y, 30);
    }
    </code>
    </div>

    function setup() {
      createCanvas(100, 100);

      // p5.Color オブジェクトを作成します。
      let c = color('deeppink');

      // オブジェクトを保存します。
      storeItem('color', c);

      describe('灰色の背景にピンクの円。');
    }

    function draw() {
      background(200);

      // オブジェクトを取得します。
      let c = getItem('color');

      // 円のスタイルを設定します。
      fill(c);

      // 円を描画します。
