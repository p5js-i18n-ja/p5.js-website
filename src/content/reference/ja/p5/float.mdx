description: >
  <p><code>String</code>を浮動小数点（小数）の<code>Number</code>に変換します。</p>

  <p><code>float()</code>は<code>'12.34'</code>のような数字に似た文字列を数値に変換します。</p>

  <p>パラメータ<code>str</code>は変換する文字列値です。例えば、<code>float('12.34')</code>を呼び出すと、

  数値<code>12.34</code>が返されます。文字列の配列が渡された場合、例えば<code>float(['12.34', '56.78'])</code>のように、

  数値の配列が返されます。</p>

  <p>注意：<code>float('giraffe')</code>のように、文字列を数値に変換できない場合、

  <code>NaN</code>（非数）値が返されます。</p>

example:

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // 文字列変数を作成
      let original = '12.3';

      // 文字列を数値に変換
      let converted = float(original);

      // 変換された値を2倍にする
      let twice = converted * 2;

      // テキストのスタイルを設定
      textAlign(CENTER, CENTER);
      textSize(12);

      // 元の値と変換された値を表示
      text(`${original} × 2 = ${twice}`, 50, 50);

      describe('灰色の背景に黒で "12.3 × 2 = 24.6" と書かれたテキスト。');
    }
    </code>

    function setup() {

      background(200);

      let original = ['60', '30', '15'];

      // 文字列を数値に変換
      let diameters = float(original);

      for (let d of diameters) {
        // 円を描画
        circle(50, 50, d);
      }

      describe('灰色の背景に3つの白い同心円。');
    }
