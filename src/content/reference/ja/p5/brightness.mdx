description: >
  <p>色の明るさの値を取得します。</p>

  <p><code>brightness()</code>は<a href="/reference/p5/p5.Color">p5.Color</a>オブジェクト、色成分の配列、

  またはCSSの色文字列からHSBの明るさの値を抽出します。</p>

  <p>デフォルトでは、<code>brightness()</code>は色のHSBの明るさを0から100の範囲で返します。

  <a href="/reference/p5/colorMode/">colorMode()</a>がHSBに設定されている場合、

  指定された範囲内で明るさの値を返します。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // HSB色を使用します。
      colorMode(HSB);

      // p5.Colorオブジェクトを作成します。
      let c = color(0, 50, 100);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // 'brightValue'を100に設定します。
      let brightValue = brightness(c);

      // 右の長方形を描画します。
      fill(brightValue);
      rect(50, 15, 35, 70);

      describe('2つの長方形。左側はサーモンピンクで、右側は白です。');
    }
