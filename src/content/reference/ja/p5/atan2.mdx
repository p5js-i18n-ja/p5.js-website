description: >
  <p>点と原点、そして正のx軸によって形成される角度を計算します。</p>

  <p><code>atan2()</code>は、主にジオメトリをマウスの位置に向けるために使用されます。
  例えば、<code>atan2(mouseY, mouseX)</code>のように使います。最初のパラメータは点の
  y座標で、2番目のパラメータはx座標です。</p>

  <p>デフォルトでは、<code>atan2()</code>は-π（約-3.14）からπ（3.14）の範囲で
  値を返します。<a href="/reference/p5/angleMode/">angleMode()</a>が<code>DEGREES</code>に
  設定されている場合、値は-180から180の範囲で返されます。</p>

    function setup() {
      createCanvas(100, 100);

      describe('キャンバスの中央にある長方形が、マウスの動きに合わせて回転します。');
    }

    function draw() {
      background(200);

      // 原点を中心に移動します
      translate(50, 50);

      // 原点に対するマウスの座標を取得します
      let x = mouseX - 50;
      let y = mouseY - 50;

      // マウスと原点の間の角度を計算します
      let a = atan2(y, x);

      // 回転させます
      rotate(a);

      // 図形を描画します
      rect(-30, -5, 60, 10);
    }
