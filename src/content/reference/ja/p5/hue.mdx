description: >
  <p>色の色相の値を取得します。</p>

  <p><code>hue()</code>は<a href="/reference/p5/p5.Color">p5.Color</a>オブジェクト、色成分の配列、

  またはCSSの色文字列から色相の値を抽出します。</p>

  <p>色相は色相環上の色の位置を表します。デフォルトでは、<code>hue()</code>は

  色のHSL色相を0から360の範囲で返します。<a href="/reference/colorMode/">colorMode()</a>が

  HSBまたはHSLに設定されている場合、指定されたモードでの色相の値を返します。</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // HSL色を使用します。
      colorMode(HSL);

      // HSL値を使用してp5.Colorオブジェクトを作成します。
      let c = color(0, 50, 100);

      // 左の長方形を描画します。
      noStroke();
      fill(c);
      rect(15, 20, 35, 60);

      // 'hueValue'を0に設定します。
      let hueValue = hue(c);

      // 右の長方形を描画します。
      fill(hueValue);
      rect(50, 20, 35, 60);

      describe(
        '2つの長方形。左側はサーモンピンクで、右側は黒です。'
      );
    }
