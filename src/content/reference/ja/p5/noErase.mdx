    function draw() {
      background(235, 145, 15);
      circle(33, 50, 60);

      // 円を部分的に消去
      erase();
      circle(67, 50, 60);
      noErase();

      // 通常の描画モードに戻る
      fill(0);
      circle(50, 75, 20);
