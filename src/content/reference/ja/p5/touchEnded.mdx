description: >
  <p>スクリーンタッチが終了するたびに1回呼び出される関数です。</p>

  <p><code>touchEnded()</code>関数を宣言すると、ユーザーがタッチスクリーンデバイスでのタッチを
  終了したときに自動的に実行されるコードブロックを設定します：</p>

  <pre><code class="language-js">function touchEnded() {
    // 実行するコード
  }

  </code></pre>

  <p><code>touchEnded()</code>がp5.jsによって呼び出されると、<a href="/reference/p5/touches/">touches</a>配列が
  最新のタッチポイントで更新されます：</p>

  <pre><code class="language-js">function touchEnded() {
    // 背景を塗りつぶす
    background(200);

    // ユーザーがタッチを終了したときに、
    // 残っている各タッチポイントをマークする
    for (let touch of touches) {
      circle(touch.x, touch.y, 40);
    }
  }

  </code></pre>

  <p>パラメータのeventはオプションです。<code>touchEnded()</code>には
  タッチイベントを説明するプロパティを持つ
  <a href="https://developer.mozilla.org/en-US/docs/Web/API/TouchEvent/"
  target="_blank">TouchEvent</a>
  オブジェクトが渡されます：</p>

  <pre><code class="language-js">function touchEnded(event) {
    // イベントを使用するコード
    console.log(event);
  }

  </code></pre>

  <p>タッチスクリーンデバイスでは、<code>touchEnded()</code>が宣言されていない場合、
  ユーザーのタッチが終了すると<a href="/reference/p5/mouseReleased/">mouseReleased()</a>が
  実行されます。<code>touchEnded()</code>が宣言されている場合、ユーザーのタッチが終了すると
  <code>touchEnded()</code>が実行され、<a href="/reference/p5/mouseReleased/">mouseReleased()</a>は
  実行されません。</p>

  <p>注意：<a href="/reference/p5/touchStarted/">touchStarted()</a>、
  <code>touchEnded()</code>、<a href="/reference/p5/touchMoved/">touchMoved()</a>は
  すべて関連しています。<a href="/reference/p5/touchStarted/">touchStarted()</a>は
  ユーザーがタッチスクリーンデバイスにタッチした瞬間に実行されます。<code>touchEnded()</code>は
  ユーザーがタッチを終了した瞬間に実行されます。<a href="/reference/p5/touchMoved/">touchMoved()</a>は
  ユーザーがタッチポイントを動かすたびに繰り返し実行されます。</p>

    let fillColor = 255;
    let borderWidth = 0.5;

    function setup() {
      createCanvas(100, 100);

      describe(
        '上部中央に数字0がある灰色の正方形。数字はユーザーが画面にタッチしている場所の数を追跡します。各タッチポイントに円が表示され、イベントに応じてスタイルが変化します。'

      textAlign(CENTER);
      textSize(16);
      fill(0);
      noStroke();

      // タッチポイントの数を表示
      text(touches.length, 50, 20);

      // タッチポイントのスタイルを設定
      fill(fillColor);
      stroke(0);
      strokeWeight(borderWidth);

      // タッチポイントを円として表示
      for (let touch of touches) {
        circle(touch.x, touch.y, 40);
      }
    }

    function touchStarted() {

    function touchEnded() {
      fillColor = random(0, 255);
    }

    // ストロークの太さを設定
    function touchMoved() {
      // ボーダーの幅を増加
      borderWidth += 0.1;

      // ボーダーの幅が大きくなりすぎたらリセット
      if (borderWidth > 20) {
        borderWidth = 0.5;
      }
    }
